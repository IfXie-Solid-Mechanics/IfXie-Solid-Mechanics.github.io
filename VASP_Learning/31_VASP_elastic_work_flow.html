    # VASP应力应变计算弹性模量工作流

    计算弹性模量的优化工作流

    # 1. 对于非合金构型，不需要超晶胞结构

    # 1.1 构建脚本

    根据经验晶格常数进行建模：

    #!/bin/bash

# 定义晶格常数，单位为埃
declare -A lattice_constants
lattice_constants[Fe]=3.450
lattice_constants[Ni]=3.516
lattice_constants[Cr]=3.625
lattice_constants[Co]=3.458
lattice_constants[Mn]=3.508
lattice_constants[Pd]=3.949

# 定义元素名称
declare -A elements
elements[Cr]="Cr"
elements[Mn]="Mn"
elements[Fe]="Fe"
elements[Co]="Co"
elements[Ni]="Ni"
elements[Pd]="Pd"

# 定义INCAR截断能量(eV)
declare -A cutoffe
cutoffe[Cr]=295.204
cutoffe[Mn]=350.8232
cutoffe[Fe]=348.3584
cutoffe[Co]=348.3584
cutoffe[Ni]=350.3916
cutoffe[Pd]=326.2025

for system in "${!elements[@]}"; do

element=$(echo ${elements[$system]})

mkdir $element
cd $element
atomsk --create fcc ${lattice_constants[$element]} $element POSCAR

cat >INCAR <<!
IBRION = 2

NPAR = 8

ISMEAR = 1
SIGMA = 0.2

ISPIN = 2

ISTART = 0
ICHARG = 2

ISIF = 7
ISYM = 0

NSW = 1000
NELM = 600

ENCUT = ${cutoffe[$element]}
PREC = High

EDIFF = 1E-4

ALGO = Fast

KSPACING = 0.1
LWAVE = .FALSE.
LCHARG  =  .FALSE.
!

cp ~/POTCAR_vasp/VASP/$element/POTCAR .

cd ..
done

    减小晶胞大小后，理论上应该对EDIFF也减小，但是由于打开磁矩后收敛困难，以及最终允许的能量误差在1meV/atom以内，使用EDIFF = 1E-4仍然可以满足精度要求。

    # 1.2 re-relax操作

    将上述结构优化的CONTCAR进行提取，并且将提取出的构型再次进行relax操作，这一步是为了防止Pulay stress的影响。

    如果re-relax的离子步小于或等于2，则这个构型已经足够精确，否则继续进行re-relax操作。

    由于离子实数目较小（4个），所以两步relax只有15分钟，不要担心在这个部分的算力消耗。如果大于两步relax，则说明这步结构再优化是有意义的，也不算浪费算力。(而且往往最多re-relax两次)

    # 2. 应变应力测量部分

    根据之前的经验来看，对能量求二阶导和对应力求一阶导时，计算出的模量误差小于1Gpa，所以方便起见，这里通过对于应力求一阶导分析模量

    # 2.1 构建xx变形脚本

    根据之前的实践经验，变形步长取2×10−4是较优的，并且在小范围内，零点的奇异性明显，因此我们需要避开这一影响，建议向左取5个数据样本。

    xx方向

    #!/bin/bash
element="Pd"
cp POSCAR POSCAR-01
for i in $(seq 1 1 5);
do
    ii=$(echo "scale=6; $i/-5000" | bc)
    mkdir $element-$(printf "%03d" $i)
    cd $element-$(printf "%03d" $i)
    cp ../POSCAR-01 .
    atomsk POSCAR-01 -sort species up -frac -deform x $ii POSCAR
    rm POSCAR-01
    cd ..
done
rm POSCAR-01

    # 2.3 应力获取脚本

    x方向应力与应变

    #!/bin/bash
element="Pd"
echo "Strain    Stress_Gpa" >> SSX.dat
for i in $(seq 5 -1 1)
do
    strain=$(echo "scale=6; $i/-5000" | bc)
    st_x=$(grep  'in kB' $element-$(printf "%03d" $i)/OUTCAR | awk '{print $3}' | tail -n 1)
    stress_x=$(echo `echo "scale=12;-0.1*$st_x" | bc |awk '{printf "%.12f",$0}'`)
    echo "$strain    $stress_x" >>  SSX.dat
done

    # 2.4 模量获取脚本

    注意根据不同模量修改SS.dat为SSX.dat,SSY.dat,SSXY.dat

    import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression

# 读取SS.dat文件
data = pd.read_csv('SS.dat', delim_whitespace=True, header=0)

# 提取应变和应力数据
strain = data['Strain'].values
stress = data['Stress_Gpa'].values

# 将应变转换为二维数组以适应线性回归模型
X = strain.reshape(-1, 1)
y = stress

# 创建并拟合线性回归模型
model = LinearRegression()
model.fit(X, y)

# 获取斜率和R平方值
slope = model.coef_[0]
r_squared = model.score(X, y)

# 输出结果，保留六位小数显示斜率和四位小数显示R²
print(f'斜率为: {slope:.6f}')
print(f'R平方值为: {r_squared:.4f}')
