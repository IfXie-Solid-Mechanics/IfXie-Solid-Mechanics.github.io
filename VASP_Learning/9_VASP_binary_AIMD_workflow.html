    # 生成VASP二元合金的POSCAR文件和AIMD其他输入文件的工作流

    主要工具为ATAT软件，sqs2poscar插件和atomsk插件。

    # 1 确定生成的dataset

    本次工作的目标势函数是FeNiCrCoPd和FeNiCrCoMn的势函数，因此总计14种元素组合的二元合金需要被建模；

    由于是3×3×3的FCC超晶胞，总计108个原子，所以每种合金的原子比例总共设计了3档，即27:81,54:54,81:27；

    对于合金的晶格常数设置，每种构型的AxB1−x合金有3种晶格常数设置，分别是aA,xaA+(1−x)aB,aB；

    对于每种合金构型，各进行400步的AIMD，各取10个snapshots。

    综上所述，二元合金总dataset的大小是14×3×3×10=1260，接近之前预设的1440左右数据量的budget。

    # 2 使用ATAT生成

    # 2.1 rndstr.in文件创建

    由于需要写14种元素组合，且每种组合有3种元素比例，所以逐个手写rndstr.in文件要写42份，所以用bash脚本生成rndstr.in文件并分类到不同的目标目录就是一种很自然的想法。

    #!/bin/bash

# 定义元素名称和比例，从行数也可以确定设置全了14种元素组合
declare -A elements
elements[FeNi]="Fe Ni"
elements[FeCo]="Fe Co"
elements[CrFe]="Cr Fe"
elements[FePd]="Fe Pd"
elements[MnFe]="Mn Fe"
elements[CrNi]="Cr Ni"
elements[CoNi]="Co Ni"
elements[NiPd]="Ni Pd"
elements[MnNi]="Mn Ni"
elements[CrCo]="Cr Co"
elements[CrPd]="Cr Pd"
elements[CrMn]="Cr Mn"
elements[CoPd]="Co Pd"
elements[MnCo]="Mn Co"

# 元素比例配置，共计3种
ratios=("0.25,0.75" "0.5,0.5" "0.75,0.25")

# 原始rndstr.in的内容模板，FCC结构
template="1.00000000 0.00000000 0.00000000
0.00000000 1.00000000 0.00000000
0.00000000 0.00000000 1.00000000
1 0 0
0 1 0
0 0 1
0.00000000 0.00000000 0.00000000 Fe=0.5, Ni=0.5
0.50000000 0.50000000 0.00000000 Fe=0.5, Ni=0.5
0.00000000 0.50000000 0.50000000 Fe=0.5, Ni=0.5
0.50000000 0.00000000 0.50000000 Fe=0.5, Ni=0.5"

# sqscell.out 的内容，333超晶胞扩胞
sqscell_content="1
3 0 0
0 3 0
0 0 3"

# 批量生成文件
for system in "${!elements[@]}"; do
  # 获取当前元素
  element_pair="${elements[$system]}"
  element1=$(echo $element_pair | cut -d' ' -f1)
  element2=$(echo $element_pair | cut -d' ' -f2)

  # 创建目录
  mkdir -p "$system"_1 "$system"_2 "$system"_3

  # 根据比例生成rndstr.in文件
  for i in ${!ratios[@]}; do
    ratio=${ratios[$i]}
    ratio1=$(echo $ratio | cut -d',' -f1)
    ratio2=$(echo $ratio | cut -d',' -f2)

    # 替换模板中的元素名称和比例
    modified_template=$(echo "$template" | sed "s/Fe=0.5, Ni=0.5/$element1=$ratio1, $element2=$ratio2/")

    # 将修改后的内容写入到对应的文件中
    echo "$modified_template" > "$system"_"$((i+1))"/rndstr.in

    # 在同一目录下写入sqscell.out文件
    echo "$sqscell_content" > "$system"_"$((i+1))"/sqscell.out
  done
done

echo "rndstr.in and sqscell.out files have been generated."

    # 2.3 mcsqs命令的运行

    到42个目录下，执行42次mcsqs命令，然后再适时停下，这同样是一个适合使用bash脚本进行的步骤。

    #!/bin/bash

# 遍历所有目录，假设目录名称包含数字后缀（如 FeNi_1, FeNi_2,...）
for dir in */; do
  # 检查目录是否为空
  if [ -d "$dir" ]; then
    echo "进入目录: $dir"

    # 进入目录
    cd "$dir" || continue

    # 执行 mcsqs -2=2 命令
    echo "执行 mcsqs -2=2..."
    mcsqs -2=2
    if [ $? -ne 0 ]; then
      echo "mcsqs -2=2 失败，跳过此目录"
      cd ..
      continue
    fi

    # 执行 mcsqs -rc 命令，并限制时间为 5 分钟
    echo "执行 mcsqs -rc (限制 5 分钟)..."
    timeout 5m mcsqs -rc
    if [ $? -eq 124 ]; then
      echo "mcsqs -rc 超过时间限制，已终止"
    fi

    # 返回上一层目录
    cd ..
  fi
done

echo "脚本执行完毕"

    但是ZHANG01可是拥有160个逻辑CPU的强大服务器！因此我们可以使用并行脚本提高42倍工作效率！让3.5小时的串行任务在5分钟的并行任务下完成。

    #!/bin/bash

# 定义一个函数来处理每个目录的任务
run_mcsqs_in_directory() {
  dir=$1
  
  # 进入目录
  cd "$dir" || return

  # 执行 mcsqs -2=2 命令
  echo "执行 mcsqs -2=2 in $dir"
  mcsqs -2=2
  if [ $? -ne 0 ]; then
    echo "mcsqs -2=2 失败，跳过此目录"
    return
  fi

  # 执行 mcsqs -rc 命令，并限制时间为 5 分钟
  echo "执行 mcsqs -rc (限制 5 分钟) in $dir"
  timeout 5m mcsqs -rc
  if [ $? -eq 124 ]; then
    echo "mcsqs -rc 超过时间限制，已终止 in $dir"
  fi

  # 返回上一层目录
  cd ..
}

# 导出函数以便 parallel 使用
export -f run_mcsqs_in_directory

# 获取所有目录
dirs=$(ls -d */)

# 使用 parallel 并行执行任务，CPU数取一个大于42的数即可，我写了80
echo "$dirs" | parallel -j 80 run_mcsqs_in_directory

    # 3 调用插件生成VASP输入文件

    因为最终步骤都需要用atomsk转化文件，并且设置超算运算脚本时需要注明在西安或者乌镇上的队列，因此我把最终文件都放在了~/05_atomsk/目录下，即~/05_atomsk/binary_xian/目录下放着要在西安上跑AIMD的构型。

    # 3.1 POSCAR文件的生成

    这一步，我们需要完成4个步骤：

    1. 调用sqs2poscar插件对于第2部分ATAT运行结果中的bestsqs.out文件转化为bestsqs.out-POSCAR文件；
    sqs2poscar bestsqs.out
    1

    1. 通过对于每个bestsqs.out-POSCAR文件创建3个目录，目录名称由当前目录名加上_1,_2,_3表示3种晶格常数；

    2. 晶格常数设置部分，首先给出基态FCC金属的晶格常数分别为Fe:3.450,Ni:3.516,Cr:3.625,Co:3.458,Mn:3.508,Pd:3.949（单位均为埃）。然后3种晶格常数的设置，以FeNi合金为例，_1目录下的是Fe的晶格常数，_2目录下的是Ni的晶格常数，_3目录下的是Fe的晶格常数与Ni的晶格常数按照元素比例取线性值。

    sed -i '2s/.*/3.450/' bestsqs.out-POSCAR
#3.450就是一个设置的晶格常数
    1
    2
    1. 最终使用atomsk对于文件bestsqs.out-POSCAR进行格式转化
    atomsk bestsqs.out-POSCAR -sort species up -frac POSCAR
    1

    综上这个bash脚本比较复杂，具体代码如下：

    #!/bin/bash

# 定义晶格常数，单位为埃
declare -A lattice_constants
lattice_constants[Fe]=3.450
lattice_constants[Ni]=3.516
lattice_constants[Cr]=3.625
lattice_constants[Co]=3.458
lattice_constants[Mn]=3.508
lattice_constants[Pd]=3.949

# 定义元素名称和比例
declare -A elements
elements[FeNi]="Fe Ni"
elements[FeCo]="Fe Co"
elements[CrFe]="Cr Fe"
elements[FePd]="Fe Pd"
elements[MnFe]="Mn Fe"
elements[CrNi]="Cr Ni"
elements[CoNi]="Co Ni"
elements[NiPd]="Ni Pd"
elements[MnNi]="Mn Ni"
elements[CrCo]="Cr Co"
elements[CrPd]="Cr Pd"
elements[CrMn]="Cr Mn"
elements[CoPd]="Co Pd"
elements[MnCo]="Mn Co"

# 元素比例配置
ratios=("0.25,0.75" "0.5,0.5" "0.75,0.25")

# 定义sqs2poscar插件和atomsk路径
SQS2POSCAR="sqs2poscar"
ATOMSK="atomsk"

# 批量处理每个元素体系
for system in "${!elements[@]}"; do
  # 获取当前元素对
  element_pair="${elements[$system]}"
  element1=$(echo $element_pair | cut -d' ' -f1)
  element2=$(echo $element_pair | cut -d' ' -f2)

  # 遍历每个比例目录（FeNi_1, FeNi_2, FeNi_3等）
  for i in ${!ratios[@]}; do
    ratio=${ratios[$i]}
    ratio1=$(echo $ratio | cut -d',' -f1)
    ratio2=$(echo $ratio | cut -d',' -f2)

    # 计算目标目录名，例如 FeNi_1, FeNi_2, FeNi_3
    dest_dir="$system"_"$((i+1))"

    cd "$dest_dir"

    # 创建对应的目录结构
    mkdir -p "$dest_dir"_1 "$dest_dir"_2 "$dest_dir"_3

    # 获取当前比例的目录
    dest_dir_1="$dest_dir"_1
    dest_dir_2="$dest_dir"_2
    dest_dir_3="$dest_dir"_3

    # 根据子目录的第二个 _1,_2,_3 来选择晶格常数
    for sub_dir in "$dest_dir_1" "$dest_dir_2" "$dest_dir_3"; do
      
      cp "bestsqs.out" "$sub_dir/bestsqs.out"

      cd "$sub_dir"

      $SQS2POSCAR "bestsqs.out"

      if [ ! -f "bestsqs.out-POSCAR" ]; then
        echo "Error: $sub_dir/bestsqs.out-POSCAR 文件不存在，跳过该目录。"
        continue
      fi

      # 获取当前子目录编号（1, 2, 3）来选择晶格常数
      sub_dir_num=$(echo "$sub_dir" | sed 's/.*_\(.*\)$/\1/') # 提取 _1, _2, _3

      # 设定对应比例的晶格常数
      if [ "$sub_dir_num" -eq 1 ]; then
        # _1目录下使用Fe的晶格常数
        lattice_const=${lattice_constants[$element1]}
      elif [ "$sub_dir_num" -eq 2 ]; then
        # _2目录下使用Ni的晶格常数
        lattice_const=${lattice_constants[$element2]}
      else
        # _3目录下使用根据比例取线性值的晶格常数
        lattice_const1=${lattice_constants[$element1]}
        lattice_const2=${lattice_constants[$element2]}
        lattice_const=$(echo "$lattice_const1 * $ratio1 + $lattice_const2 * $ratio2" | bc)
      fi

      # 修改晶格常数到bestsqs.out-POSCAR文件
      sed -i "2s/.*/$lattice_const/" "bestsqs.out-POSCAR"

      # 调用atomsk进行格式转换
      echo "使用 atomsk 转换 $sub_dir/bestsqs.out-POSCAR 文件"
      $ATOMSK "bestsqs.out-POSCAR" -sort species up -frac "POSCAR"

      echo "$sub_dir/bestsqs.out-POSCAR 文件处理完成！"

      cd ..
    done
    cd ..
  done
done

echo "所有处理完成！"
