      # 2. 获得VASP输入数据

      对于周边滑移简易代码如下所示，基于Atomsk的bash脚本:

      #!/bin/bash
atomsk POSCAR test.cfg
# 循环480次生成480个新的POSCAR文件
for i in $(seq 1 1 480); do
ii=$(echo "$i * 0.003" | bc)
mkdir test_$(printf "%03d" $i)
cd test_$(printf "%03d" $i)
cp ../test.cfg .
atomsk test.cfg -shift above 11 z 0 $ii 0 POSCAR
mv POSCAR ../POSCAR_$(printf "%03d" $i)
cd ..
rm -r test_$(printf "%03d" $i)
done

      然后对于中心原子进行较大步长的移动，扫描：

      #!/bin/bash
atomsk POSCAR test.cfg
for i in $(seq 1 1 47); do
    ii=$(echo "$i * 0.03" | bc)
    iii=$((48-$i))
    mkdir result_$(printf "%03d" $i)
    cd result_$(printf "%03d" $i)
    for j in $(seq 1 1 $iii); do
        mkdir test_$(printf "%03d" $j)
        cd test_$(printf "%03d" $j)
        cp ../../test.cfg .
        y=$(echo "$ii - $j * 0.03 * 0.5" | bc)
        x=$(echo "$j * 0.03 * 0.866" | bc)
        atomsk test.cfg -shift above 11 z $x $y 0 POSCAR
        mv POSCAR ../POSCAR_$(printf "%03d" $j)
        cd ..
        rm -r test_$(printf "%03d" $j)
    done
    cd ..
done

      为了提高计算的并行效率，需要将480个POSCAR_文件分为40个一组，分为12组计算

      #!/bin/bash

# 循环40个文件一组进行处理
for i in {1..12}
do
  # 计算文件的起始编号和结束编号
  start=$(( (i - 1) * 40 + 1 ))   # 计算每组的起始文件编号
  end=$(( i * 40 ))                # 计算每组的结束文件编号
  
  # 创建目标目录（如果没有的话）
  result_dir="./result_$(printf "%02d" $i)"
  mkdir -p "$result_dir"
  
  # 打包文件到对应的目录中
  for j in $(seq -f "%03g" $start $end)
  do
    # 将每个文件移动到相应目录中
    mv "POSCAR_$j" "$result_dir/"
  done
done

      scnet上计算脚本

      #!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

# Dirrectory containing the jobs
WORK_DIR=$(pwd)

# Paths to the INCAR and POTCAR files
INCAR_PATH="$WORK_DIR/../INCAR"
POTCAR_PATH="$WORK_DIR/../POTCAR"

# List of job directories
JOB_DIRS=$(ls POSCAR_*)

for JOB_DIR in $JOB_DIRS; do
    # Go into the job directory
    cd $WORK_DIR
    DIR_NAME="dir_$JOB_DIR"
    mkdir $DIR_NAME
    cd $WORK_DIR/$DIR_NAME

    # Create symbolic links to the INCAR and POTCAR files
    ln -s $INCAR_PATH INCAR
    ln -s $POTCAR_PATH POTCAR
    ln -s $WORK_DIR/$JOB_DIR POSCAR

    srun --mpi=pmi2 vasp_std
done

      import matplotlib.pyplot as plt

# 生成x和y坐标
x_coords = []
y_coords = []

for i in range(1, 48):  # i从1到47
    ii = i * 0.03
    iii = 48 - i
    for j in range(1, iii + 1):
        x = ii - j * 0.03 * 0.5
        y = j * 0.03 * 0.866
        x_coords.append(x)
        y_coords.append(y)

# 读取z值数据
with open('result_stacking.txt', 'r') as f:
    z_values = [float(line.strip()) for line in f]

# 确保数据长度一致
assert len(x_coords) == len(y_coords) == len(z_values), "Data lengths are inconsistent."

# 绘制平面云图
plt.scatter(x_coords, y_coords, c=z_values, cmap='viridis', s=20)
plt.colorbar(label='Total Energy(eV)')  # 添加颜色条，显示值的范围

# 设置图形属性
plt.xlabel('X [11-2]')
plt.ylabel('Y [1-10]')
plt.title('Stacking fault gamma surface')
plt.grid(True)  # 显示网格线
plt.axis('equal')  # 确保x和y轴比例相同

# 可选：保存图表
plt.savefig('energy_plot.png', dpi=300, bbox_inches='tight')
