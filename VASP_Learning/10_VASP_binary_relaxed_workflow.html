      # VASP 二元合金基于relaxed structure的工作流设计

      根据张老师在2025.01.07的建议，需要考虑给二元合金增加relaxed structure，并且以此为基础来设计晶格常数也更加合理。

      # 1. 确定生成的dataset

      本次工作的目标势函数是FeNiCrCoPd和FeNiCrCoMn的势函数，因此总计14种元素组合的二元合金需要被建模；

      由于是3×3×3的FCC超晶胞，总计108个原子，所以每种合金的原子比例总共设计了3档，即27:81,54:54,81:27；

      对于上述42种构型，寻找relaxed structure晶格常数；

      每种relaxed structure和体应变运行400步的AIMD，各取10个snapshots；

      综上所述，补充的二元合金dataset大小是14×3×10=420个。

      # 2. 确定晶格常数

      # 2.1 目录结构

      我们在之前已经计算出了3档合金在aA,xaA+(1−x)aB,aB晶格常数下的能量，而基于线弹性理论，我们近似地认为能量是晶格常数的二次函数，因此根据上述3个样本进行二次函数的最小二乘拟合。（对于二元合金其实是直接进行系数求解）

      需要用python脚本分别读取到3个dir_POSCAR-001/目录下的VASP计算信息，之后再进行二次函数最小二乘拟合，将读取信息需要存储到与FeNi_result/目录同级的FeNi_1_relaxed/中，而python脚本放在FeNi_result/目录的父目录下,下面是python实现代码：

      import os
import re
import numpy as np
from scipy.optimize import curve_fit

# 定义二次函数，用于最小二乘拟合
def quadratic(x, a, b, c):
    return a * x**2 + b * x + c

# 读取OSZICAR文件中的F值
def read_oszicar_energy(file_path):
    energy_values = []
    with open(file_path, 'r') as file:
        for line in file:
            if "F=" in line:
                # 使用正则表达式提取 F= 后的数值
                match = re.search(r"F=\s*(-?\.\d+E[+-]?\d+)", line)
                if match:
                    f_value = float(match.group(1))
                    energy_values.append(f_value)
    return energy_values

# 读取POSCAR文件中的晶格常数
def read_poscar_lattice_constant(file_path):
    with open(file_path, 'r') as file:
        # 跳过前两行
        for _ in range(2):
            file.readline()

        # 读取第三行
        line = file.readline().strip()
        # 分割字符串并获取第一列数据
        third_line_values = line.split()
        first_column_value = float(third_line_values[0])

        # 返回晶格常数
        return first_column_value / 3

# 遍历FeNi_result目录中的子目录，读取OSZICAR和POSCAR文件
def process_directory():
    # 目录路径
    root_dir = 'FeNi_result/'
    result_dir = 'FeNi_1_relaxed/'
    
    if not os.path.exists(result_dir):
        os.makedirs(result_dir)

    energy_values = []  # 用来存储 3 个样本点的能量
    lattice_constants = []  # 用来存储 3 个样本点的晶格常数

    # 遍历每个子目录，读取一个样本点
    for subdir in ['result_FeNi_1_1', 'result_FeNi_1_2', 'result_FeNi_1_3']:
        subdir_path = os.path.join(root_dir, subdir)
        
        if os.path.isdir(subdir_path):
            # 处理每个result_FeNi_X_X目录下的dir_POSCAR-001目录
            poscar_dir = os.path.join(subdir_path, 'dir_POSCAR-001')
            if os.path.exists(poscar_dir):
                oszicar_path = os.path.join(poscar_dir, 'OSZICAR')
                poscar_path = os.path.join(poscar_dir, 'POSCAR')
                
                # 读取OSZICAR中的能量数据
                energy_values += read_oszicar_energy(oszicar_path)[:1]  # 只取第一个样本点
                # 读取POSCAR中的晶格常数
                lattice_constants.append(read_poscar_lattice_constant(poscar_path))  # 取晶格常数

    # 确保至少读取了 3 个样本点
    if len(energy_values) >= 3 and len(lattice_constants) >= 3:
        # 使用最小二乘法进行二次函数拟合
        params, covariance = curve_fit(quadratic, lattice_constants, energy_values)
        
        # 拟合后的参数
        a, b, c = params
        
        # 计算二次函数的最小值及对应的x值
        min_x = -b / (2 * a)
        min_y = quadratic(min_x, *params)

        # 输出拟合结果到文件
        result_file = os.path.join(result_dir, "fit_results.txt")
        with open(result_file, 'a') as result_f:
            result_f.write(f"Fit Parameters: a={a}, b={b}, c={c}\n")
            result_f.write(f"Energy Values: {energy_values}\n")
            result_f.write(f"Lattice Constants: {lattice_constants}\n")
            result_f.write(f"Minimum Energy: {min_y}, at x = {min_x}\n\n")
        print(f"Fitted parameters and minimum value saved to {result_file}")
    else:
        print("Error: Not enough data points to fit the quadratic function.")

if __name__ == "__main__":
    process_directory()

      # 3.1 生成输入文件

      两个脚本，作用参考上一篇文章；

      #!/bin/bash

# 定义元素名称和比例
declare -A elements
elements[FeNi]="Fe Ni"
elements[FeCo]="Fe Co"
elements[CrFe]="Cr Fe"
elements[FePd]="Fe Pd"
elements[MnFe]="Mn Fe"
elements[CrNi]="Cr Ni"
elements[CoNi]="Co Ni"
elements[NiPd]="Ni Pd"
elements[MnNi]="Mn Ni"
elements[CrCo]="Cr Co"
elements[CrPd]="Cr Pd"
elements[CrMn]="Cr Mn"
elements[CoPd]="Co Pd"
elements[MnCo]="Mn Co"

# 元素比例配置
ratios=("0.25,0.75" "0.5,0.5" "0.75,0.25")

# 定义sqs2poscar插件和atomsk路径
SQS2POSCAR="sqs2poscar"
ATOMSK="atomsk"

# 批量处理每个元素体系
for system in "${!elements[@]}"; do
  # 获取当前元素对
  element_pair="${elements[$system]}"
  element1=$(echo $element_pair | cut -d' ' -f1)
  element2=$(echo $element_pair | cut -d' ' -f2)

  # 遍历每个比例目录（FeNi_1, FeNi_2, FeNi_3等）
  for i in ${!ratios[@]}; do
    # 计算目标目录名，例如 FeNi_1, FeNi_2, FeNi_3
    dest_dir="$system"_"$((i+1))"

    cd "$dest_dir"

    # 创建对应的目录结构
    mkdir ${dest_dir}_relaxed

    # 获取当前比例的目录
    dest_dir_1=${dest_dir}_relaxed

    cp "bestsqs.out" "$dest_dir_1/bestsqs.out"

    cd "$dest_dir_1"

    $SQS2POSCAR "bestsqs.out"

    # 读取文件并提取 "x = " 后的数字
    result_file="$HOME/fasttrans/tmp/${element1}${element2}_$((i+1))_relaxed/fit_results.txt"

    # 确保文件存在
    if [ -f "$result_file" ]; then
      # 使用 grep 和 awk 提取 x 后的数字
      x_value=$(grep -oP 'at x = \K[0-9.-]+' "$result_file")
      echo "Extracted x_value: $x_value"
    else
      echo "Error: $result_file does not exist!"
    fi

    # 使用提取的 x_value 更新晶格常数
    lattice_const=$x_value

    # 修改晶格常数到bestsqs.out-POSCAR文件
    sed -i "2s/.*/$lattice_const/" "bestsqs.out-POSCAR"

    # 调用atomsk进行格式转换
    echo "使用 atomsk 转换 $dest_dir_1/bestsqs.out-POSCAR 文件"
    $ATOMSK "bestsqs.out-POSCAR" -sort species up -frac "POSCAR"

    cd ..
    cd ..
  done
done

echo "所有处理完成！"

      # 4. 计算数据收集

      计算数据收集直接在原有一元金属的基础上进行了改进：

      1. 需要区分什么时候收集CrMnFeCoNi,什么时候收集CrFeCoNiPd，因此写了两个sh脚本；

      2. 在一元的金属上直接复制粘贴，然后修改目录名称即可。

      3. 未来可以尝试一种直接从更高一级目录搜索的写法。（但是这样对于数据的收集整理有更高要求，从04_vasp_data下面还需要增加一层）

      所以copy_outcar_CrFeCoNiPd.sh脚本如下：

      #!/bin/bash

# 启用globstar以支持**/递归匹配
shopt -s globstar

# 定义目标目录，用来存放所有的OUTCAR文件
output_dir="./all_outcar_files_CrFeCoNiPd"
mkdir -p $output_dir

# 初始化计数器
count=0

# 遍历Fe_training_data目录下的所有子目录，查找OUTCAR文件
for dir in ./Fe_training_data/**/; do
  if [ -f "$dir/OUTCAR" ]; then
    echo "Found OUTCAR in: $dir"
    
    # 调试输出，检查路径
    echo "Original directory path: $dir"
    
    # 去掉路径中的 './' 前缀，然后替换 '/' 为 '_'
    safe_name=$(echo "$dir" | sed 's|^\./||' | tr '/' '_')
    echo "Safe file name: $safe_name"
    
    # 复制文件
    cp "$dir/OUTCAR" "$output_dir/${safe_name}_OUTCAR"
    
    # 增加计数
    count=$((count + 1))
  fi
done
