      # VASP五元合金堆垛层错构型补丁-2

      在将我自己之前的结果和gamma-surface的结果进比较时，我发现我之前的工作流少考虑了一个三角形：

      所以我们这里将进行整个gamma-surface的扫描，至少是需要扫描一个平行四边形。但是为了后期处理方便，我们这里可以选择直接扫描整个[1-10]和[11-2]平面，得到gamma-surface

      # 1. 输入构型的代码修改

      #!/bin/bash
atomsk POSCAR test.cfg
for i in $(seq 1 1 41); do
    ii=$(echo "$i * 0.125 - 0.125" | bc)
    mkdir result_$(printf "%03d" $i)
    cd result_$(printf "%03d" $i)
    for j in $(seq 1 1 41); do
        mkdir test_$(printf "%03d" $j)
        cd test_$(printf "%03d" $j)
        cp ../../test.cfg .
	jj=$(echo "$j * 0.216 - 0.216" | bc)
        x=$(echo "$ii" | bc)
        y=$(echo "$jj" | bc)
        atomsk test.cfg -shift above 11 z $x $y 0 POSCAR
        mv POSCAR ../POSCAR_$(printf "%03d" $j)
        cd ..
        rm -r test_$(printf "%03d" $j)
    done
    cd ..
done

      # 2. 结果收集代码

      #!/bin/bash
# 循环41个文件一组进行处理
for i in $(seq 1 1 41);
do
  result_dir="./result_$(printf "%03d" $i)"
  cd "$result_dir"
  for dir in */
  do
    cd $dir
    grep F= OSZICAR | tail -n 1 | awk '{print $3}' >> ../../result_stacking.txt
    cd ..
  done
  cd ..
done

      import matplotlib.pyplot as plt

# 生成x和y坐标
x_coords = []
y_coords = []

for i in range(1, 42):  # i从1到41
    for j in range(1, 42):
        x = i * 0.025 - 0.025
        y = j * 0.025 - 0.025
        x_coords.append(x)
        y_coords.append(y)

# 读取z值数据
with open('result_stacking.txt', 'r') as f:
    z_values = [float(line.strip()) for line in f]

# 确保数据长度一致
assert len(x_coords) == len(y_coords) == len(z_values), "Data lengths are inconsistent."

# 将x, y, z分别写入result.txt文件中，每列对应一个值
with open('result.txt', 'w') as f:
    for x, y, z in zip(x_coords, y_coords, z_values):
        line = f"{x} {y} {z}\n"
        f.write(line)
