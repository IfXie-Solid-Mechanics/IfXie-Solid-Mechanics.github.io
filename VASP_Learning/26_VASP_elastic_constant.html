# 1. 计算思路

# 1.1 输入文件生成脚本

采用高精度结构优化后的构型，参考上一篇工作，采用下面方法生成脚本

#!/bin/bash

cp POSCAR POSCAR-01 
for i in $(seq 1 1 100);
do
    #le:less than or equal
    if [ $i -le 50 ]; then
            ii=$(echo "scale=3; $i/1000" | bc)
    else
            tmp=$(($i - 50))
            ii=$(echo "scale=3; $tmp/-1000" | bc)
    fi
    mkdir CrFeCoNiPd-$(printf "%03d" $i)
    cd CrFeCoNiPd-$(printf "%03d" $i)
    cp ../POSCAR-01 .
    atomsk POSCAR-01 -sort species up -frac -deform x $ii POSCAR
    rm POSCAR-01
    cd ..
done

rm POSCAR-01

# 附录1 提取应力应变曲线代码

首先，从CrFeCoNiPd-001到CrFeCoNiPd-030文件夹内，分别表示应变从0.01增长到0.30。然后提取应力的bash命令为：

#针对6个应力分量提取
grep 'in kB' CrFeCoNiPd-030/OUTCAR
#针对x方向应力提取
grep 'in kB' CrFeCoNiPd-030/OUTCAR | awk '{printf "%12.8f \n", $3}'

综上所述，提取脚本如下所示（注意VASP的单位是kBar，因此用0.1Gpa=1Kbar进行单位换算）：

#!/bin/bash
echo "Strain    Stress_Gpa" >> SS.dat
for i in $(seq 100 -1 51);
do
        strain=$(echo "scale=3; $i/-1000" | bc)
        st_x=$(grep 'in kB' CrFeCoNiPd-$(printf "%03d" $i)/OUTCAR | awk '{printf "%12.8f \n", $3}')
        stress_x=$(echo `echo "scale=4;-0.1*$st_x" | bc |awk '{printf "%.4f",$0}'`)
        echo "$strain    $stress_x" >>  SS.dat
done

for i in $(seq 1 1 50)
do
        strain=$(echo "scale=2; $i/100" | bc)
        st_x=$(grep 'in kB' CrFeCoNiPd-$(printf "%03d" $i)/OUTCAR | awk '{printf "%12.8f \n", $3}')
        stress_x=$(echo `echo "scale=4;-0.1*$st_x" | bc |awk '{printf "%.4f",$0}'`)
        echo "$strain    $stress_x" >>  SS.dat
done

下面是绘图脚本：

import numpy as np
import matplotlib.pyplot as plt

# 读取文件数据（假设文件名为 SS.dat）
data = np.genfromtxt("SS.dat", delimiter=None, skip_header=1)
strain = data[:, 0]    
stress = data[:, 1]    

# 绘制应力-应变曲线
plt.figure()
plt.plot(strain, stress, 'b-', label='Stress-Strain Curve')
plt.xlabel('Strain')      
plt.ylabel('Stress (GPa)') 
plt.title('Stress vs. Strain')    
plt.grid(True)               
plt.legend()                 

plt.savefig("stress_strain_curve.png")
