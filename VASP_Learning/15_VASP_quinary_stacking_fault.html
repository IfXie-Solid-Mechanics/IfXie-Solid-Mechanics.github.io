# 1.2 用VASP获得准随机高熵合金构型

这一步由2个部分组成：打乱元素序列得到新的构型、用VASP做结构优化。

打乱元素顺序的bash脚本示例如下所示：

#!/bin/bash

# 输入文件名
input_file="POSCAR"
# 输出文件的前缀
output_prefix="POSCAR"
# 用于记录生成的哈希值
declare -A generated_hashes

# 获取POSCAR文件的前8行
head -n 8 "$input_file" > header.txt

# 获取POSCAR文件的第9到第56行
sed -n '9,56p' "$input_file" > body.txt

# 循环150次生成150个新的POSCAR文件
for i in {1..150}; do
    # 创建一个新的文件名
    output_file="${output_prefix}-$(printf "%03d" $i)"

    while true; do
        # 随机打乱body.txt的内容
        shuffled_body=$(shuf body.txt)

        # 计算打乱后的内容的哈希值
        hash_value=$(echo "$shuffled_body" | sha256sum | cut -d ' ' -f1)

        # 检查哈希值是否已存在
        if [[ -z ${generated_hashes[$hash_value]} ]]; then
            # 如果没有重复，保存该哈希值并退出循环
            generated_hashes[$hash_value]=1
            break
        fi
    done

    # 将前8行和打乱后的内容合并，输出到新的文件
    echo -e "$(cat header.txt)\n$shuffled_body" > "$output_file"
done

# 删除临时文件
rm header.txt
rm body.txt

echo "生成完毕！"

# 1.3 运用Atomsk生成Stacking faults构型

我于2025年2月19日进行了生成思路的优化，首先生成480个步长较短的分解位错，沿着[1-10]方向滑动，每次步长为0.003埃，最终滑移1.44埃，FCC晶格常数处以根号6。

对于周边滑移简易代码如下所示，基于Atomsk的bash脚本:

#!/bin/bash
atomsk POSCAR test.cfg
# 循环480次生成480个新的POSCAR文件
for i in $(seq 1 1 480); do
ii=$(echo "$i * 0.003" | bc)
mkdir test_$(printf "%03d" $i)
cd test_$(printf "%03d" $i)
cp ../test.cfg .
atomsk test.cfg -shift above 11 z $ii 0 0 POSCAR
mv POSCAR ../POSCAR_$(printf "%03d" $i)
cd ..
rm -r test_$(printf "%03d" $i)
done

#!/bin/bash

for i in $(seq 1 1 47); do
    ii=$(echo "$i * 0.03" | bc)
    iii=$((48-$i))
    mkdir result_$(printf "%03d" $i)
    cd result_$(printf "%03d" $i)
    for j in $(seq 1 1 $iii); do
        mkdir test_$(printf "%03d" $j)
        cd test_$(printf "%03d" $j)
        cp ../../test.cfg .
        x=$(echo "$ii - $j * 0.03 * 0.5" | bc)
        y=$(echo "$j * 0.03 * 0.866" | bc)

        mv POSCAR ../POSCAR_$(printf "%03d" $j)
        cd ..
        rm -r test_$(printf "%03d" $j)
    done
    cd ..
done

但是为了提高计算的并行效率进一步榨干scnet的算力，需要将480个POSCAR_文件分为40个一组，分为12组计算

#!/bin/bash

# 循环40个文件一组进行处理
for i in {1..12}
do
  # 计算文件的起始编号和结束编号
  start=$(( (i - 1) * 40 + 1 ))   # 计算每组的起始文件编号
  end=$(( i * 40 ))                # 计算每组的结束文件编号
  
  # 创建目标目录（如果没有的话）
  result_dir="./result_$(printf "%02d" $i)"
  mkdir -p "$result_dir"
  
  # 打包文件到对应的目录中
  for j in $(seq -f "%03g" $start $end)
  do
    # 将每个文件移动到相应目录中
    mv "POSCAR_$j" "$result_dir/"
  done
done

#!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

# Dirrectory containing the jobs
WORK_DIR=$(pwd)

# Paths to the INCAR and POTCAR files
INCAR_PATH="$WORK_DIR/../INCAR"
POTCAR_PATH="$WORK_DIR/../POTCAR"

# List of job directories
JOB_DIRS=$(ls POSCAR_*)

for JOB_DIR in $JOB_DIRS; do
    # Go into the job directory
    cd $WORK_DIR
    DIR_NAME="dir_$JOB_DIR"
    mkdir $DIR_NAME
    cd $WORK_DIR/$DIR_NAME

    # Create symbolic links to the INCAR and POTCAR files
    ln -s $INCAR_PATH INCAR
    ln -s $POTCAR_PATH POTCAR
    ln -s $WORK_DIR/$JOB_DIR POSCAR

    srun --mpi=pmi2 vasp_std
done

我于2025年2月19日进行了生成思路的优化，首先生成480个步长较短的分解位错，沿着[1-10]方向滑动，每次步长为0.003埃，最终滑移1.44埃，FCC晶格常数处以根号6。

对于周边滑移简易代码如下所示，基于Atomsk的bash脚本:

#!/bin/bash
atomsk POSCAR test.cfg
# 循环480次生成480个新的POSCAR文件
for i in $(seq 1 1 480); do
ii=$(echo "$i * 0.003" | bc)
mkdir test_$(printf "%03d" $i)
cd test_$(printf "%03d" $i)
cp ../test.cfg .
atomsk test.cfg -shift above 11 z $ii 0 0 POSCAR
mv POSCAR ../POSCAR_$(printf "%03d" $i)
cd ..
rm -r test_$(printf "%03d" $i)
done

#!/bin/bash
atomsk POSCAR test.cfg
for i in $(seq 1 1 47); do
    ii=$(echo "$i * 0.03" | bc)
    iii=$((48-$i))
    mkdir result_$(printf "%03d" $i)
    cd result_$(printf "%03d" $i)
    for j in $(seq 1 1 $iii); do
        mkdir test_$(printf "%03d" $j)
        cd test_$(printf "%03d" $j)
        cp ../../test.cfg .
        x=$(echo "$ii - $j * 0.03 * 0.5" | bc)
        y=$(echo "$j * 0.03 * 0.866" | bc)
        atomsk test.cfg -shift above 11 z $x $y 0 POSCAR
        mv POSCAR ../POSCAR_$(printf "%03d" $j)
        cd ..
        rm -r test_$(printf "%03d" $j)
    done
    cd ..
done

但是为了提高计算的并行效率进一步榨干scnet的算力，需要将480个POSCAR_文件分为40个一组，分为12组计算

#!/bin/bash

# 循环40个文件一组进行处理
for i in {1..12}
do
  # 计算文件的起始编号和结束编号
  start=$(( (i - 1) * 40 + 1 ))   # 计算每组的起始文件编号
  end=$(( i * 40 ))                # 计算每组的结束文件编号
  
  # 创建目标目录（如果没有的话）
  result_dir="./result_$(printf "%02d" $i)"
  mkdir -p "$result_dir"
  
  # 打包文件到对应的目录中
  for j in $(seq -f "%03g" $start $end)
  do
    # 将每个文件移动到相应目录中
    mv "POSCAR_$j" "$result_dir/"
  done
done

而在scnet上，对于cal.slurm也需要适用特殊提交脚本，以便于高效地并行

#!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

# Dirrectory containing the jobs
WORK_DIR=$(pwd)

# Paths to the INCAR and POTCAR files
INCAR_PATH="$WORK_DIR/../INCAR"
POTCAR_PATH="$WORK_DIR/../POTCAR"

# List of job directories
JOB_DIRS=$(ls POSCAR_*)

for JOB_DIR in $JOB_DIRS; do
    # Go into the job directory
    cd $WORK_DIR
    DIR_NAME="dir_$JOB_DIR"
    mkdir $DIR_NAME
    cd $WORK_DIR/$DIR_NAME

    # Create symbolic links to the INCAR and POTCAR files
    ln -s $INCAR_PATH INCAR
    ln -s $POTCAR_PATH POTCAR
    ln -s $WORK_DIR/$JOB_DIR POSCAR

    srun --mpi=pmi2 vasp_std
done
