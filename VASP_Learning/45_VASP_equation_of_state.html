          # 状态方程的对比

          # 1. 寻找状态方程

          寻找状态方程的方法通常是：直接在论文中查找（比较大海捞针）；直接去materials project上面查找（正确性存疑）；去一些其他数据开源平台查找（正确性更加存疑）

          # 2. 对于Ni的fcc构型

          在materials project找到了一个结果(https://legacy.materialsproject.org/materials/mp-23/)

          然后通过之前工作中提到的状态方程计算代码，计算Ni-fcc的状态方程，和结果进行对比。绘图代码在附录1中，计算代码在附录2中，得到的初步结果如下所示

          可以看出在体积较小侧，能量变化迅速，拟合效果不是很好，所以我们需要加密我们的样本点再进行一次对比，结果如下（把附录2中的步长从0.05改0.025即可）：

          直接观察结果具有不错的相似性。如果需要比较严格误差的话，可以将附录2中代码调整一下，使得对应体积点一一对应，然后减去一个平均误差后比较。

          # 附录1 python绘图代码

          import json
import matplotlib.pyplot as plt
import numpy as np

# 读取JSON文件数据
with open('Ni-fcc.json', 'r') as f:
    json_data = json.load(f)

# 提取JSON数据并排序
volumes_json = json_data['volumes']
energies_json = json_data['energies']

# 将数据组合并排序（按体积从小到大）
combined_json = sorted(zip(volumes_json, energies_json), key=lambda x: x[0])
volumes_json_sorted = [item[0] for item in combined_json]
energies_json_sorted = [item[1] for item in combined_json]

# 读取ev.out文件数据
x_ev = []  # 存储第二列 (V0*4)
y_ev = []  # 存储第三列 (free_energy)

with open('ev.out', 'r') as file:
    lines = file.readlines()
    # 跳过第一行标题行
    for line in lines[1:]:
        # 分割每行数据并移除空值
        data = line.split()
        if len(data) >= 3:  # 确保行中有足够的数据
            try:
                x_ev.append(float(data[1]))   # 第一列转换为浮点数
                y_ev.append(float(data[2]))   # 第三列转换为浮点数
            except ValueError:
                print(f"警告：跳过无法解析的行: {line.strip()}")

# 转换ev.out数据并排序
for i in range(len(x_ev)):
    x_ev[i] = x_ev[i] / 4
    y_ev[i] = y_ev[i] / 4

# 对ev.out数据排序
combined_ev = sorted(zip(x_ev, y_ev), key=lambda x: x[0])
x_ev_sorted = [item[0] for item in combined_ev]
y_ev_sorted = [item[1] for item in combined_ev]

# 创建图形
plt.figure(figsize=(12, 8), dpi=100)

# 绘制排序后的JSON数据曲线
plt.plot(volumes_json_sorted, energies_json_sorted, 
         'o-', 
         linewidth=2, 
         markersize=8,
         markerfacecolor='blue',
         markeredgecolor='darkblue',
         color='royalblue',
         alpha=0.8,
         label='Energy(Reference)')

# 绘制排序后的ev.out数据曲线
plt.plot(x_ev_sorted, y_ev_sorted, 
         's-', 
         linewidth=2, 
         markersize=8,
         markerfacecolor='red',
         markeredgecolor='darkred',
         color='crimson',
         alpha=0.8,
         label='Energy(mine)')

# 添加标签和标题
plt.xlabel('Cell Volume (AA^3)', fontsize=14)
plt.ylabel('Energy (eV/atom)', fontsize=14)
plt.title('Equation of State(E-V)', fontsize=16, fontweight='bold')
plt.grid(True, linestyle='--', alpha=0.7)

# 自动调整坐标轴范围
all_x = volumes_json_sorted + x_ev_sorted
all_y = energies_json_sorted + y_ev_sorted
plt.xlim(min(all_x)*0.98, max(all_x)*1.02)
plt.ylim(min(all_y)*1.05, max(all_y)*0.95) if min(all_y) < 0 else plt.ylim(min(all_y)*0.95, max(all_y)*1.05)

# 添加图例
plt.legend(fontsize=12, loc='best')

# 突出显示最小值点
min_index_json = energies_json_sorted.index(min(energies_json_sorted))
min_index_ev = y_ev_sorted.index(min(y_ev_sorted))

plt.scatter(volumes_json_sorted[min_index_json], energies_json_sorted[min_index_json], 
            s=150, c='blue', edgecolor='black', zorder=5, marker='*', label='Min Energy (Reference)')
plt.scatter(x_ev_sorted[min_index_ev], y_ev_sorted[min_index_ev], 
            s=150, c='red', edgecolor='black', zorder=5, marker='*', label='Min Energy (mine)')

# 添加图例并调整位置
plt.legend(fontsize=10, loc='upper right')

# 添加网格和美化布局
plt.grid(True, linestyle=':', alpha=0.7)
plt.tight_layout()

          # 附录2 VASP计算代码

          cat > INCAR.relax <<!
Global Parameters
  NPAR   = 8             (Max is no. nodes; don't set for hybrids)
  ISTART =  0            (Read existing wavefunction; if there)
  ICHARG =  2            (Non-self-consistent: GGA/LDA band structures)
  LREAL  = F             (Projection operators: automatic)
  ENCUT  =  500          (Cut-off energy for plane wave basis set, in eV) 
  PREC   =  Accurate     (Precision level)  
  LWAVE  = .FALSE.       (Write WAVECAR or not)
  LCHARG = .FALSE.       (Write CHGCAR or not) 
  ADDGRID= .TRUE.        (Increase grid; helps GGA convergence) 
  ISPIN = 2              (spin-polarized calculations are performed.)
  # LVTOT  = .TRUE.      (Write total electrostatic potential into LOCPOT or not)
  # LVHAR  = .TRUE.      (Write ionic + Hartree electrostatic potential into LOCPOT or not)
  # NELECT =             (No. of electrons: charged cells; be careful)
  # LPLANE = .TRUE.      (Real space distribution; supercells)

Electronic Relaxation
  ISMEAR =  1            (Gaussian smearing; metals:1)
  SIGMA  =  0.20         (Smearing value in eV; metals:0.2)
  NELM   =  60           (Max electronic SCF steps)  
  NELMIN =  4            (Min electronic SCF steps)
  EDIFF  =  1E-08        (SCF energy convergence; in eV) 
  # GGA  =  PE           (PBEsol exchange-correlation)

Ionic Relaxation
  NELMIN =  6            (Min electronic SCF steps) 
  NSW    =  60           (Max electronic SCF steps)
  IBRION =  2            (Algorithm: 0-MD; 1-Quasi-New; 2-CG)
  ISIF   =  4            (Stress/relaxation: 2-Ions, 3-Shape/Ions/V, 4-Shape/Ions)
  EDIFFG = -0.001        (Ionic convergence; eV/AA)
  # ISYM =  2            (Symmetry: 0=none; 2=GGA; 3=hybrids)
!

cat > INCAR.static <<!
Global Parameters
  NPAR   = 8             (Max is no. nodes; don't set for hybrids)
  ISTART =  0            (Read existing wavefunction; if there)
  ICHARG =  2            (Non-self-consistent: GGA/LDA band structures)
  LREAL  = F             (Projection operators: automatic)
  ENCUT  =  500          (Cut-off energy for plane wave basis set, in eV) 
  PREC   =  Accurate     (Precision level)  
  LWAVE  = .FALSE.       (Write WAVECAR or not)
  LCHARG = .FALSE.       (Write CHGCAR or not) 
  ADDGRID= .TRUE.        (Increase grid; helps GGA convergence) 
  ISPIN = 2              (spin-polarized calculations are performed.)
  # LVTOT  = .TRUE.      (Write total electrostatic potential into LOCPOT or not)
  # LVHAR  = .TRUE.      (Write ionic + Hartree electrostatic potential into LOCPOT or not)
  # NELECT =             (No. of electrons: charged cells; be careful)
  # LPLANE = .TRUE.      (Real space distribution; supercells)
  # GGA   = PE

Static Calculation
  ISMEAR = -5            (tetrahedron method for DOS) 
  # LORBIT =  11         (PAW radii for projected DOS)
  # NEDOS  =  2001       (DOSCAR points) 
  NELM   =  60           (Max electronic SCF steps)
  EDIFF  =  1E-08        (SCF energy convergence; in eV)
  EDIFFG = -0.001
!

cat > KPOINTS <<!
Automatic generation
0
Monkhorst-pack
10  10  10
0  0  0
!

echo 'a0' 'volume' 'free_energy(eV)' >ev.out

for i in $(seq 3.15 0.05 3.85)
do

cat > POSCAR <<!
Ni4 fcc
1.0000000000
    $i                  0.0000000000        0.0000000000
    0.0000000000        $i                  0.0000000000
    0.0000000000        0.0000000000        $i
Ni
4
Direct
    0.0000000000        0.0000000000        0.0000000000
    0.5000000000        0.0000000000        0.5000000000
    0.0000000000        0.5000000000        0.5000000000
    0.5000000000        0.5000000000        0.0000000000
!

#优化计算
cp INCAR.relax INCAR
echo "a=$i"; time mpirun -n 64  ~/vasp.6.4.1/bin/vasp_std
cp CONTCAR POSCAR
rm INCAR

#静态计算
cp INCAR.static INCAR
echo "a=$i"; time mpirun -n 64  ~/vasp.6.4.1/bin/vasp_std
V=$(grep "volume" OUTCAR | tail -1 | awk '{printf "%12.9f \n", $5 }')
E=$(grep "TOTEN" OUTCAR | tail -1 | awk '{printf "%12.9f \n", $5 }')
echo $i $V $E >> ev.out
rm INCAR
done
