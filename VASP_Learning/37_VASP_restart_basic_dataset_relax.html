      # 2. 五元合金构型扩充

      对于五元合金，我们不仅计算了原本的SQSs构型，而且还各自增加了4组原子排列被随机打乱的构型，使得最终有10组五元合金初始构型被纳入计算，符合改进版本的工作流要求，扩充了对于五元合金的相空间分析。打乱代码如下所示：

      #!/bin/bash

# 输入文件名
input_file="POSCAR"
# 输出文件的前缀
output_prefix="POSCAR"
# 用于记录生成的哈希值
declare -A generated_hashes

# 获取POSCAR文件的前8行
head -n 8 "$input_file" > header.txt

# 获取POSCAR文件的第9到第116行
sed -n '9,116p' "$input_file" > body.txt

# 循环4次生成4个新的POSCAR文件
for i in {1..4}; do
    # 创建一个新的文件名
    output_file="${output_prefix}-$(printf "%03d" $i)"

    while true; do
        # 随机打乱body.txt的内容
        shuffled_body=$(shuf body.txt)

        # 计算打乱后的内容的哈希值
        hash_value=$(echo "$shuffled_body" | sha256sum | cut -d ' ' -f1)

        # 检查哈希值是否已存在
        if [[ -z ${generated_hashes[$hash_value]} ]]; then
            # 如果没有重复，保存该哈希值并退出循环
            generated_hashes[$hash_value]=1
            break
        fi
    done

    # 将前8行和打乱后的内容合并，输出到新的文件
    echo -e "$(cat header.txt)\n$shuffled_body" > "$output_file"
done

# 删除临时文件
rm header.txt
rm body.txt

echo "生成完毕！"
