# 1.1 AIMD运行

对于9种四元合金的re-relaxed构型用于在3个温度下运行AIMD，运行AIMD的效率较高，因为最终将对AIMD每40个step进行截取，并进行更高精度的计算。

#!/bin/bash

for i in $(seq 300 1000 2300);do
target_path=~/08_basic_dataset_AIMD/quaternary_wuzhen_$(printf "%04d" $i)
mkdir $target_path

cat >INCAR_AIMD <<!
IBRION = 0

NPAR = 8

ISMEAR = 1
SIGMA = 0.2

ISPIN = 1

ISIF = 0
ISYM = 0

NSW = 1600
POTIM = 1.0

NELM = 600

MDALGO = 2
SMASS = 0

TEBEG = $i
TEEND = $i

PREC = Normal

EDIFF = 1E-4

ALGO = VeryFast

KSPACING = 0.8
LWAVE = .FALSE.
LCHARG  =  .FALSE. 
!

for file in */;do
cd $file
mkdir $target_path/$file

for subfile in */;do
cd $subfile
mkdir $target_path/${file}${subfile}
cp CONTCAR $target_path/${file}${subfile}POSCAR
cp POTCAR $target_path/${file}${subfile}POTCAR
cp ../../INCAR_AIMD $target_path/${file}${subfile}INCAR
cp cal.slurm $target_path/${file}${subfile}cal.slurm
cd ..
done

cd ..
done

rm INCAR_AIMD
done

# 1.2 AIMD后处理

调用python的ASE库实现从AIMD的XDATCAR结果中提取出新的POSCAR，具体的extract_config.py和convert_AIMD_snapshots.sh脚本如下

#!/usr/bin/env python

## Import relevant library
from ase.io import read, write

## Input parameters
start = 40
interval = 40
nblock = 1

## Read all trajectory from OUTCAR
traj = read('XDATCAR', index='::1', format='vasp-xdatcar')  ## read AIMD trajectory, interval is 1;
print('Read all {} trajectory structures'.format(len(traj)))

## Write interval configurations in POSCAR
num_extract = 0
for i in range(int(start/nblock)-1, len(traj), int(interval/nblock)):
    num_extract += 1
    write('POSCAR-{:03d}'.format(num_extract), traj[i], format='vasp', direct=True)
print('Extract {} trajectory structures'.format(num_extract))

在超算上的任务提交文件也需要做一些针对性的修改

#!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

# Dirrectory containing the jobs
WORK_DIR=$(pwd)

# Paths to the INCAR and POTCAR files
INCAR_PATH="/work/home/if_xie/10_basic_dataset_AIMD_snapshots/INCAR"
POTCAR_PATH="$WORK_DIR/POTCAR"

# List of job directories
JOB_DIRS=$(ls POSCAR-*)

for JOB_DIR in $JOB_DIRS; do
    # Go into the job directory
    cd $WORK_DIR
    DIR_NAME="dir_$JOB_DIR"
    mkdir $DIR_NAME
    cd $WORK_DIR/$DIR_NAME

    # Create symbolic links to the INCAR and POTCAR files
    ln -s $INCAR_PATH INCAR
    ln -s $POTCAR_PATH POTCAR
    ln -s $WORK_DIR/$JOB_DIR POSCAR

    srun --mpi=pmi2 vasp_std
done

# 2.1 应变构型获取

对于一个bulk的应变类型，分为单轴拉压（控制其余方向应变为0），纯剪切变形，体应变。

与二元合金的工作流相比，这里仅仅修改了target_path变量

#!/bin/bash

for i in $(seq 1 1 3);do

if [ "$i" -eq 1 ]; then
    target_path=~/09_basic_dataset_strained/quaternary_strained_xx
    mkdir $target_path

    for element_path in */;do
    cd $element_path
    mkdir ${target_path}/$element_path

    for file in */;do
    cd $file
    cp CONTCAR POSCAR-01
    mkdir large-tensile-$file
    cd large-tensile-$file

    for k in $(seq 1 1 10);do
    mkdir $(printf "%03d" $k)-$file
    cd $(printf "%03d" $k)-$file
    kk=$(echo "scale=6; $k/100" | bc)
    cp ../../POSCAR-01 .
    cp ../../POTCAR .
    atomsk POSCAR-01 -sort species up -frac -deform x $kk POSCAR
    rm POSCAR-01
    cd ..
    done

    cd ..
    mkdir small-tensile-$file
    cd small-tensile-$file

    mkdir $(printf "%03d" $k)-$file
    cd $(printf "%03d" $k)-$file
    kk=$(echo "scale=6; $k/10000" | bc)
    cp ../../POSCAR-01 .
    cp ../../POTCAR .
    atomsk POSCAR-01 -sort species up -frac -deform x $kk POSCAR
    rm POSCAR-01
    cd ..
    done

    cd ..
    mkdir large-compressive-$file
    cd large-compressive-$file

    for k in $(seq 1 1 10);do
    mkdir $(printf "%03d" $k)-$file
    cd $(printf "%03d" $k)-$file
    kk=$(echo "scale=6; $k/-100" | bc)
    cp ../../POSCAR-01 .
    cp ../../POTCAR .
    atomsk POSCAR-01 -sort species up -frac -deform x $kk POSCAR
    rm POSCAR-01
    cd ..
    done

    cd ..
    mkdir small-compressive-$file
    cd small-compressive-$file

    for k in $(seq 1 1 10);do
    mkdir $(printf "%03d" $k)-$file
    cd $(printf "%03d" $k)-$file
    kk=$(echo "scale=6; $k/-10000" | bc)
    cp ../../POSCAR-01 .
    cp ../../POTCAR .
    atomsk POSCAR-01 -sort species up -frac -deform x $kk POSCAR
    rm POSCAR-01
    cd ..
    done

    cd ..
    mv large-tensile-$file $target_path/$element_path
    mv small-tensile-$file $target_path/$element_path
    mv large-compressive-$file $target_path/$element_path
    mv small-compressive-$file $target_path/$element_path
    rm POSCAR-01
    cd ..
    done

    cd ..
    done
elif [ "$i" -eq 2 ]; then
    target_path=~/09_basic_dataset_strained/quaternary_strained_xy

    for element_path in */;do
    cd $element_path

    cd $file

    mkdir $(printf "%03d" $k)-$file

    cp ../../POSCAR-01 .
    cp ../../POTCAR .
    atomsk POSCAR-01 -sort species up -frac -deform xy $kk POSCAR
    rm POSCAR-01
    cd ..

    cd ..

    mkdir small-shearing-$file

    for k in $(seq 1 1 10);do
    mkdir $(printf "%03d" $k)-$file
    cd $(printf "%03d" $k)-$file
    kk=$(echo "scale=6; $k/10000" | bc)
    cp ../../POSCAR-01 .
    cp ../../POTCAR .
    atomsk POSCAR-01 -sort species up -frac -deform xy $kk POSCAR
    rm POSCAR-01
    cd ..
    done
    cd ..

    mv large-shearing-$file ${target_path}/$element_path
    mv small-shearing-$file ${target_path}/$element_path
    rm POSCAR-01
    cd ..
    done

    cd ..
    done
elif [ "$i" -eq 3 ]; then
    target_path=~/09_basic_dataset_strained/quaternary_strained_bulk
    mkdir $target_path

    for element_path in */;do
    cd $element_path
    mkdir ${target_path}/$element_path

# 2.2 应变构型的结构优化计算

提交计算任务的思路是按照每个合金的应变类型来提交的，所以总计提交 9×8=72个计算任务。

所以每个计算任务下有10个及以下的构型，最终计算结果可能超过400，最终计算时间在48小时以内。

slurm文件如下所示：

#!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显


for JOB_DIR in */; do
    cd $JOB_DIR
    ln -s ../../../INCAR INCAR
    srun --mpi=pmi2 vasp_std
    cd ..
done
