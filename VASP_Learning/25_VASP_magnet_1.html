# 附录1. 能量提取脚本

读取OSZICAR文件

import os
import re
import numpy as np

result_dir = './'
energy_values = []
with open('OSZICAR', 'r') as file:
    for line in file:
        if "F=" in line:
                # 使用正则表达式提取 F= 后的数值
            match = re.search(r"F=\s*(-?\.\d+E[+-]?\d+)", line)
            if match:
                f_value = float(match.group(1))
                energy_values.append(f_value)
result_file = os.path.join(result_dir, "energy_result.txt")
with open(result_file, 'w') as result_f:
    for value in energy_values:
        result_f.write(f"{value:.8f}\n")
print(f"Energy value saved to {result_file}")

# 附录3. 构型提取脚本

运行下面这个脚本后取最后一个构型

#!/usr/bin/env python

## Import relevant library
from ase.io import read, write

## Input parameters
start = 1
interval = 1
nblock = 1

## Read all trajectory from OUTCAR
traj = read('XDATCAR', index='::1', format='vasp-xdatcar')  ## read AIMD trajectory, interval is 1;
print('Read all {} trajectory structures'.format(len(traj)))

## Write interval configurations in POSCAR
num_extract = 0
for i in range(int(start/nblock)-1, len(traj), int(interval/nblock)):
    num_extract += 1
    write('POSCAR-{:03d}'.format(num_extract), traj[i], format='vasp', direct=True)
print('Extract {} trajectory structures'.format(num_extract))
