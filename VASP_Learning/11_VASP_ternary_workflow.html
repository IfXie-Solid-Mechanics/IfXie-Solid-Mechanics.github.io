          # VASP三元合金整合工作流

          主要工具为ATAT软件，sqs2poscar插件和atomsk插件。

          # 1 确定生成的dataset

          本次工作的目标势函数是FeNiCrCoPd和FeNiCrCoMn的势函数，因此总计16种元素组合的三元合金需要被建模；

          每种合金的原子比例总仅设计1档，即36:36:36；

          对于合金的晶格常数设置，每种构型的AxABxBCxC合金有4种晶格常数设置，分别是aA,aB,aC,xAaA+xBaB+xCaC；

          对于每种合金构型，各进行400步的AIMD，各取10个snapshots。

          综上所述，三元合金初始dataset的大小是16×1×4×10=640，由于后续还有对于平衡态的变形分析，因此这里仅为三元合金的部分数据。

          # 2 使用ATAT生成

          # 2.1 rndstr.in文件创建

          由于需要写16种元素组合，且每种组合有1种元素比例，所以逐个手写rndstr.in文件要写16份，所以用bash脚本生成rndstr.in文件并分类到不同的目标目录就是一种很自然的想法。

          #!/bin/bash

# 定义元素名称和比例，从行数也可以确定设置全了16种元素组合
declare -A elements
elements[CoNiPd]="Co Ni Pd"
elements[CrFeNi]="Cr Fe Ni"
elements[CrFeCo]="Cr Fe Co"
elements[CrFePd]="Cr Fe Pd"
elements[CrMnFe]="Cr Mn Fe"
elements[CrCoNi]="Cr Co Ni"
elements[CrNiPd]="Cr Ni Pd"
elements[CrMnNi]="Cr Mn Ni"
elements[CrCoPd]="Cr Co Pd"
elements[CrMnCo]="Cr Mn Co"
elements[FeCoNi]="Fe Co Ni"
elements[FeNiPd]="Fe Ni Pd"
elements[FeCoPd]="Fe Co Pd"
elements[MnFeNi]="Mn Fe Ni"
elements[MnFeCo]="Mn Fe Co"
elements[MnCoNi]="Mn Co Ni"

# 元素比例配置，共计1种
ratios=("0.3333333,0.3333333,0.3333333")

# 原始rndstr.in的内容模板，FCC结构
template="1.00000000 0.00000000 0.00000000
0.00000000 1.00000000 0.00000000
0.00000000 0.00000000 1.00000000
1 0 0
0 1 0
0 0 1
0.00000000 0.00000000 0.00000000 Fe=0.33, Ni=0.33, Cr=0.34
0.50000000 0.50000000 0.00000000 Fe=0.33, Ni=0.33, Cr=0.34
0.00000000 0.50000000 0.50000000 Fe=0.33, Ni=0.33, Cr=0.34
0.50000000 0.00000000 0.50000000 Fe=0.33, Ni=0.33, Cr=0.34"

# sqscell.out 的内容，333超晶胞扩胞
sqscell_content="1
3 0 0
0 3 0
0 0 3"

# 批量生成文件
for system in "${!elements[@]}"; do
  # 获取当前元素
  element_pair="${elements[$system]}"
  element1=$(echo $element_pair | cut -d' ' -f1)
  element2=$(echo $element_pair | cut -d' ' -f2)
  element3=$(echo $element_pair | cut -d' ' -f3)

  # 创建目录
  mkdir -p "$system"_1

  # 根据比例生成rndstr.in文件
  for i in ${!ratios[@]}; do
    ratio=${ratios[$i]}
    ratio1=$(echo $ratio | cut -d',' -f1)
    ratio2=$(echo $ratio | cut -d',' -f2)
    ratio3=$(echo $ratio | cut -d',' -f3)

    # 替换模板中的元素名称和比例
    modified_template=$(echo "$template" | sed "s/Fe=0.33, Ni=0.33, Cr=0.34/$element1=$ratio1, $element2=$ratio2, $element3=$ratio3/")

    # 将修改后的内容写入到对应的文件中
    echo "$modified_template" > "$system"_"$((i+1))"/rndstr.in

    # 在同一目录下写入sqscell.out文件
    echo "$sqscell_content" > "$system"_"$((i+1))"/sqscell.out
  done
done

echo "rndstr.in and sqscell.out files have been generated."

          #!/bin/bash

# 定义一个函数来处理每个目录的任务
run_mcsqs_in_directory() {
  dir=$1
  
  # 进入目录
  cd "$dir" || return

  # 执行 mcsqs -2=2 命令
  echo "执行 mcsqs -2=2 in $dir"
  mcsqs -2=2
  if [ $? -ne 0 ]; then
    echo "mcsqs -2=2 失败，跳过此目录"
    return
  fi

  # 执行 mcsqs -rc 命令，并限制时间为 5 分钟
  echo "执行 mcsqs -rc (限制 5 分钟) in $dir"
  timeout 5m mcsqs -rc
  if [ $? -eq 124 ]; then
    echo "mcsqs -rc 超过时间限制，已终止 in $dir"
  fi

  # 返回上一层目录
  cd ..
}

# 导出函数以便 parallel 使用
export -f run_mcsqs_in_directory

# 获取所有目录
dirs=$(ls -d */)

# 使用 parallel 并行执行任务，CPU数取一个大于16的数即可，我写了80
echo "$dirs" | parallel -j 80 run_mcsqs_in_directory

          综上这个bash脚本比较复杂，具体代码如下：

          #!/bin/bash

# 定义晶格常数，单位为埃
declare -A lattice_constants
lattice_constants[Fe]=3.450
lattice_constants[Ni]=3.516
lattice_constants[Cr]=3.625
lattice_constants[Co]=3.458
lattice_constants[Mn]=3.508
lattice_constants[Pd]=3.949

# 定义元素名称和比例，从行数也可以确定设置全了16种元素组合
declare -A elements
elements[CoNiPd]="Co Ni Pd"
elements[CrFeNi]="Cr Fe Ni"
elements[CrFeCo]="Cr Fe Co"
elements[CrFePd]="Cr Fe Pd"
elements[CrMnFe]="Cr Mn Fe"
elements[CrCoNi]="Cr Co Ni"
elements[CrNiPd]="Cr Ni Pd"
elements[CrMnNi]="Cr Mn Ni"
elements[CrCoPd]="Cr Co Pd"
elements[CrMnCo]="Cr Mn Co"
elements[FeCoNi]="Fe Co Ni"
elements[FeNiPd]="Fe Ni Pd"
elements[FeCoPd]="Fe Co Pd"
elements[MnFeNi]="Mn Fe Ni"
elements[MnFeCo]="Mn Fe Co"
elements[MnCoNi]="Mn Co Ni"

# 元素比例配置，共计1种
ratios=("0.3333333,0.3333333,0.3333333")

# 定义sqs2poscar插件和atomsk路径
SQS2POSCAR="sqs2poscar"
ATOMSK="atomsk"

# 批量处理每个元素体系
for system in "${!elements[@]}"; do
  # 获取当前元素对
  element_pair="${elements[$system]}"
  element1=$(echo $element_pair | cut -d' ' -f1)
  element2=$(echo $element_pair | cut -d' ' -f2)
  element3=$(echo $element_pair | cut -d' ' -f3)

  # 遍历每个比例目录（FeNi_1, FeNi_2, FeNi_3等）
  for i in ${!ratios[@]}; do
    ratio=${ratios[$i]}
    ratio1=$(echo $ratio | cut -d',' -f1)
    ratio2=$(echo $ratio | cut -d',' -f2)
    ratio3=$(echo $ratio | cut -d',' -f3)

    # 计算目标目录名，例如 FeNi_1, FeNi_2, FeNi_3
    dest_dir="$system"_"$((i+1))"

    cd "$dest_dir"

    # 创建对应的目录结构
    mkdir -p "$dest_dir"_1 "$dest_dir"_2 "$dest_dir"_3 "$dest_dir"_4

    # 获取当前比例的目录
    dest_dir_1="$dest_dir"_1
    dest_dir_2="$dest_dir"_2
    dest_dir_3="$dest_dir"_3
    dest_dir_4="$dest_dir"_4

    # 根据子目录的第二个 _1,_2,_3 来选择晶格常数
    for sub_dir in "$dest_dir_1" "$dest_dir_2" "$dest_dir_3" "$dest_dir_4"; do
      
      cp "bestsqs.out" "$sub_dir/bestsqs.out"

      cd "$sub_dir"

      $SQS2POSCAR "bestsqs.out"

      if [ ! -f "bestsqs.out-POSCAR" ]; then
        echo "Error: $sub_dir/bestsqs.out-POSCAR 文件不存在，跳过该目录。"
        continue
      fi

      # 获取当前子目录编号（1, 2, 3, 4）来选择晶格常数
      sub_dir_num=$(echo "$sub_dir" | sed 's/.*_\(.*\)$/\1/') # 提取 _1, _2, _3, _4

      # 设定对应比例的晶格常数
      if [ "$sub_dir_num" -eq 1 ]; then
        # _1目录下使用Fe的晶格常数
        lattice_const=${lattice_constants[$element1]}
      elif [ "$sub_dir_num" -eq 2 ]; then
        # _2目录下使用Ni的晶格常数
        lattice_const=${lattice_constants[$element2]}
      elif [ "$sub_dir_num" -eq 3 ]; then
        # _2目录下使用Ni的晶格常数
        lattice_const=${lattice_constants[$element3]}
      else
        # _3目录下使用根据比例取线性值的晶格常数
        lattice_const1=${lattice_constants[$element1]}
        lattice_const2=${lattice_constants[$element2]}
        lattice_const3=${lattice_constants[$element3]}
        lattice_const=$(echo "$lattice_const1 * $ratio1 + $lattice_const2 * $ratio2 + $lattice_const3 * $ratio3" | bc)
      fi

      # 修改晶格常数到bestsqs.out-POSCAR文件
      sed -i "2s/.*/$lattice_const/" "bestsqs.out-POSCAR"

      # 调用atomsk进行格式转换
      echo "使用 atomsk 转换 $sub_dir/bestsqs.out-POSCAR 文件"
      $ATOMSK "bestsqs.out-POSCAR" -sort species up -frac "POSCAR"

      echo "$sub_dir/bestsqs.out-POSCAR 文件处理完成！"

      cd ..
    done
    cd ..
  done
done

echo "所有处理完成！"

import os

# 定义二次函数，用于最小二乘拟合
def quadratic(x, a, b, c):
    return a * x**2 + b * x + c

# 读取OSZICAR文件中的F值
def read_oszicar_energy(file_path):
    energy_values = []
    with open(file_path, 'r') as file:
        for line in file:
            if "F=" in line:
                # 使用正则表达式提取 F= 后的数值
                match = re.search(r"F=\s*(-?\.\d+E[+-]?\d+)", line)
                if match:

def read_poscar_lattice_constant(file_path):
    with open(file_path, 'r') as file:
        # 跳过前两行

        third_line_values = line.split()
        first_column_value = float(third_line_values[0])

        # 返回晶格常数
        return first_column_value / 3
