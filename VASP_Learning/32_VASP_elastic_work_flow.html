# VASP高精度应变工作流设计

参考上一篇工作思路，将千分之一以内的应变范围拓展到十分之二的量级范围。

# 1. 对于非合金构型，不需要超晶胞结构

而在之前的工作中，我们已经得到所有单质构型的含磁性结构优化构型。因此在本篇工作中，将利用之前的构型作为初始输入。

# 1.1 训练数据集设计

扩充变形数据时，我认为变形步长取5×10−3是较优的。而最终应变达到百分之二十，所以将xx和xy的压缩和拉伸构型都考虑在内时，每个单质元素有120个计算任务，总计2个单质(Ni和Pd，其他均不是fcc基态，不予考虑)，总数据量为至少240个数据，大约可以达到接近300的数据量级。

xx方向压缩

#!/bin/bash
element="Pd"
cp POSCAR POSCAR-01
for i in $(seq 1 1 40);
do
    ii=$(echo "scale=6; $i/-200" | bc)
    mkdir $element-$(printf "%03d" $i)
    cd $element-$(printf "%03d" $i)
    cp ../POSCAR-01 .
    atomsk POSCAR-01 -sort species up -frac -deform x $ii POSCAR
    rm POSCAR-01
    cd ..
done
rm POSCAR-01

# 2.2 INCAR脚本

#!/bin/bash

# 定义元素名称
declare -A elements
elements[Cr]="Cr"
elements[Mn]="Mn"
elements[Fe]="Fe"
elements[Co]="Co"
elements[Ni]="Ni"
elements[Pd]="Pd"

# 定义INCAR截断能量(eV)
declare -A cutoffe
cutoffe[Cr]=295.204
cutoffe[Mn]=350.8232
cutoffe[Fe]=348.3584
cutoffe[Co]=348.3584
cutoffe[Ni]=350.3916
cutoffe[Pd]=326.2025

for system in "${!elements[@]}"; do

element=$(echo ${elements[$system]})

cd $element

cat >INCAR <<!
IBRION = 2

NPAR = 8

ISMEAR = 1
SIGMA = 0.2

ISPIN = 2

ISTART = 0
ICHARG = 2

ISIF = 2
ISYM = 0

NSW = 1000
NELM = 600

ENCUT = ${cutoffe[$element]}
PREC = High

EDIFF = 1E-4

ALGO = Fast

KSPACING = 0.1
LWAVE = .FALSE.
LCHARG  =  .FALSE.
!

cp ~/POTCAR_vasp/VASP/$element/POTCAR .

cd ..
done
