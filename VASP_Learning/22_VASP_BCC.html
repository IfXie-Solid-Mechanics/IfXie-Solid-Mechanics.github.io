        # VASP五元合金的BCC构型整合工作流

        主要工具为ATAT软件，sqs2poscar插件和atomsk插件。

        # 1. 确定生成的dataset

        本次工作的目标势函数是FeNiCrCoPd和FeNiCrCoMn的势函数，因此总计2种元素组合的四元合金需要被建模；

        每种合金的原子比例总仅设计1档，即20:20:20:20:20；

        对于晶格常数的设置，保证BOX尺寸在结构优化中不变，而不断改变尺寸进行结构优化计算。

        然后对于结构优化的结果，还可以进行AIMD扩充数据集。

        使用插件将bestsqs.out文件转化为POSCAR文件：

        #!/bin/bash
sqs2poscar bestsqs.out
num=1
for i in $(seq 2.05 0.05 5.00);
do
    mkdir CrFeCoNiPd_$(printf "%03d" $num)
    cd CrFeCoNiPd_$(printf "%03d" $num)
    cp ../bestsqs.out-POSCAR .
    sed -i 2s/.*/${i}/ bestsqs.out-POSCAR
    atomsk bestsqs.out-POSCAR -sort species up -frac POSCAR
    cd ..
    num=$(($num+1))
done

        # 3. 使用VASP对于上述构型进行结构优化处理

        INCAR文件示例：

        # structural optimization
IBRION = 2

# Ncore = 64
NPAR = 8

# Fermi surface set
ISMEAR = 1
SIGMA = 0.2

# cal magnet
ISPIN = 2

# cal the stress
ISIF = 2

# not to read the WAVECAR file.
ISTART = 0

# VASP does not use symmetry
ISYM = 0

# the max ions & element steps
NSW = 1
NELM = 600

# the cal prec
PREC = High

# the end cut
EDIFF = 1E-4

# the algo's choice
ALGO = Fast

# the kpoint set
KSPACING = 0.2

# 1.3 E_cut of Mn
ENCUT = 350.8232

# not create WAVECAR CHGCAR
LWAVE = .FALSE.
LCHARG  =  .FALSE.
