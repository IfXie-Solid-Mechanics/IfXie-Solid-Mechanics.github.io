        # 2 KSPACING收敛性测试

        1. 计算脚本文件cal.slurm（grep命令的信息我都检查过）
        #! /bin/bash
#SBATCH -J vasp
#SBATCH -N 1
#SBATCH --ntasks-per-node=64
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

rm kspace_energy.out
for i in $(seq 0.5 -0.05 0.1) 
do
rm WAVECAR
cat >INCAR <<!
ISMEAR = 1; 
SIGMA = 0.2;

MAGMOM = 108*2
ISPIN = 2
EDIFF = 1E-4

NCORE = 2 up to number-of-cores-per-socket
KSPACING = $i
!
echo "KSPACING = $i" ; srun --mpi=pmi2 vasp_std
E=$(grep "TOTEN" OUTCAR | tail -1 | awk '{printf "%12.9f \n", $5 }')
T=$(grep "Total CPU time" OUTCAR | tail -1 | awk '{printf "%12.9f \n", $6 }')
M=$(grep "mag" OSZICAR | tail -1 | awk '{printf "%12.9f \n", $10 }')
echo $i $E $T $M>>kspace_energy.out
echo $(grep "generate k-points for" OUTCAR )>>kspace_num.out
done
