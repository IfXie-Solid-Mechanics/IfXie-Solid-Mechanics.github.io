    # VASP: 对势计算工作流

    对势分为AA对势和AB对势，针对CrFeCoNiPd和CrMnFeCoNi高熵合金，AA对势有6种，AB对势有14种。

    计划从两个原子距离0.05埃为起点，并以0.05埃为步长计算对势，上限为9.0埃，对于每种配对总计180个任务。

    总计3600个任务。

    # 1. 初始构型的构造

    首先需要构造一个足够大的box，然后将两个原子之间的距离不断拉长。

    #! /bin/bash
element1="Ni"
element2="Pd"
for i in $(seq 1 1 180); do
x1=15.50
x2=$(echo "scale=3; 15.5+$i*0.05" | bc)
cat >POSCAR_$(printf "%03d" $i) <<!
pair-potential:
 1.0
 40.0 0.0 0.0
 0.0 20.0 0.0
 0.0 0.0 20.0
 $element1 $element2
 1 1 
cartesian
$x1 10.0 10.0
$x2 10.0 10.0
!
done

    其次，出于对于计算效率的考量，可以将10个构型打包为一个目录下进行计算：

    #!/bin/bash

# 循环10个文件一组进行处理
for i in {1..18}
do
  # 计算文件的起始编号和结束编号
  start=$(( (i - 1) * 10 + 1 ))   # 计算每组的起始文件编号
  end=$(( i * 10 ))                # 计算每组的结束文件编号
  
  # 创建目标目录（如果没有的话）
  result_dir="./result_$(printf "%02d" $i)"
  mkdir -p "$result_dir"
  
  # 打包文件到对应的目录中
  for j in $(seq -f "%03g" $start $end)
  do
    # 将每个文件移动到相应目录中
    mv "POSCAR_$j" "$result_dir/"
  done
done

    # 1.1 脚本构造单质构型

    总共6种单质元素，分别是Cr, Mn, Fe, Co, Ni, Pd，所以构造脚本如下所示

    #!/bin/bash
declare -A elements
elements[Cr]="Cr"
elements[Mn]="Mn"
elements[Fe]="Fe"
elements[Co]="Co"
elements[Ni]="Ni"
elements[Pd]="Pd"

for system in "${!elements[@]}"; do
element=$(echo ${elements[$system]})
mkdir $element$element-pair-potential
cd $element$element-pair-potential

for i in $(seq 1 1 180); do
x1=15.50
x2=$(echo "scale=3; 15.5+$i*0.05" | bc)
cat >POSCAR_$(printf "%03d" $i) <<!
pair-potential:
 1.0
 40.0 0.0 0.0
 0.0 20.0 0.0
 0.0 0.0 20.0
 $element
 2 
cartesian
$x1 10.0 10.0
$x2 10.0 10.0
!
done
cp ~/POTCAR_vasp/VASP/$element/POTCAR .
cp ../pack_up.sh .
source pack_up.sh
cd ..

done

    # 1.2 脚本构造合金

    总共14种二元合金，如下所示：

    #!/bin/bash
declare -A elements
elements[FeNi]="Fe Ni"
elements[FeCo]="Fe Co"
elements[CrFe]="Cr Fe"
elements[FePd]="Fe Pd"
elements[MnFe]="Mn Fe"
elements[CrNi]="Cr Ni"
elements[CoNi]="Co Ni"
elements[NiPd]="Ni Pd"
elements[MnNi]="Mn Ni"
elements[CrCo]="Cr Co"
elements[CrPd]="Cr Pd"
elements[CrMn]="Cr Mn"
elements[CoPd]="Co Pd"
elements[MnCo]="Mn Co"

for system in "${!elements[@]}"; do
element_pair="${elements[$system]}"
element1=$(echo $element_pair | cut -d' ' -f1)
element2=$(echo $element_pair | cut -d' ' -f2)
mkdir $element1$element2-pair-potential
cd $element1$element2-pair-potential

for i in $(seq 1 1 180); do
x1=15.50
x2=$(echo "scale=3; 15.5+$i*0.05" | bc)
cat >POSCAR_$(printf "%03d" $i) <<!
pair-potential:
 1.0
 40.0 0.0 0.0
 0.0 20.0 0.0
 0.0 0.0 20.0
 $element1 $element2
 1 1
cartesian
$x1 10.0 10.0
$x2 10.0 10.0
!
done
cat ~/POTCAR_vasp/VASP/$element1/POTCAR ~/POTCAR_vasp/VASP/$element2/POTCAR > ./POTCAR
cp ../pack_up.sh .
source pack_up.sh
cd ..

done

    # structural optimization

# the algo's choice

    # 2.2 scnet批量任务提交指令

    #!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

# Dirrectory containing the jobs
WORK_DIR=$(pwd)

# Paths to the INCAR and POTCAR files
INCAR_PATH="$WORK_DIR/../../INCAR"
POTCAR_PATH="$WORK_DIR/../POTCAR"

# List of job directories
JOB_DIRS=$(ls POSCAR_*)

for JOB_DIR in $JOB_DIRS; do
    # Go into the job directory
    cd $WORK_DIR
    DIR_NAME="dir_$JOB_DIR"
    mkdir $DIR_NAME
    cd $WORK_DIR/$DIR_NAME

    # Create symbolic links to the INCAR and POTCAR files
    ln -s $INCAR_PATH INCAR
    ln -s $POTCAR_PATH POTCAR
    ln -s $WORK_DIR/$JOB_DIR POSCAR

    srun --mpi=pmi2 vasp_std
done

    # 3. 结果提取脚本

    从OSZICAR文件中提取总势能即为对势，序号的0.05倍就是两者之间的距离，提取脚本如下所示：

    #!/bin/bash

for dir in */;
do
    cd $dir
    for dirdir in */;
    do
	cd $dirdir
	grep F= OSZICAR | tail -n 1 | awk '{print $3}' >> ../../pair_potential.txt 
    	cd ..
    done
    cd ..
done
