    # VASP：FCC堆垛层错高精度计算

    # 1. 高精度计算规划

    计算构型:Ni-fcc,Pd-fcc,CrFeCoNiPd-fcc,CrMnFeCoNi-fcc

    计算任务：在1/6[112]方向上的stacking_fault，每种构型480个步长，总计1920个计算结果

    # 2. Ni-fcc和Pd-fcc的计算

    # 2.1 初始构型的建立：

    先使用Atomsk建立一个fcc晶胞，其中的晶格常数根据之前高精度relaxed构型结果设置。

    atomsk --create fcc 3.5183159685144951 Ni orient [1-10] [11-2] [111] -dup 1 2 4 POSCAR
    1

    再进入POSCAR文件中，手动修改z方向的box大小，使得滑移面法向拥有真空层，建议真空层高度为15埃。

    然后再使用Atomsk进行堆垛层错的建模

    #!/bin/bash
atomsk POSCAR test.cfg
# 循环480次生成480个新的POSCAR文件
for i in $(seq 1 1 480); do
ii=$(echo "$i * 0.003" | bc)
mkdir test_$(printf "%03d" $i)
cd test_$(printf "%03d" $i)
cp ../test.cfg .
atomsk test.cfg -shift above 11 z 0 $ii 0 POSCAR
mv POSCAR ../POSCAR_$(printf "%03d" $i)
cd ..
rm -r test_$(printf "%03d" $i)
done

    #!/bin/bash

# 循环10个文件一组进行处理
for i in {1..48}
do
  # 计算文件的起始编号和结束编号
  start=$(( (i - 1) * 10 + 1 ))   # 计算每组的起始文件编号
  end=$(( i * 10 ))                # 计算每组的结束文件编号
  
  # 创建目标目录（如果没有的话）
  result_dir="./result_$(printf "%02d" $i)"
  mkdir -p "$result_dir"
  
  # 打包文件到对应的目录中
  for j in $(seq -f "%03g" $start $end)
  do
    # 将每个文件移动到相应目录中
    mv "POSCAR_$j" "$result_dir/"
  done
done

    # 2.2 INCAR文件准备

    对于Ni和Pd需要在ENCUT处，根据POTCAR中的阶段能量进行1.3倍的增大。

    # structural optimization
IBRION = 2

# Ncore = 64
NPAR = 8

# Fermi surface set
ISMEAR = 1
SIGMA = 0.2

# cal magnet

# cal the stress

# not to read the WAVECAR file.
ISTART = 0

# VASP does not use symmetry
ISYM = 0

# the max ions & element steps
NSW = 1
NELM = 600

# the cal prec
PREC = High

# the end cut
EDIFF = 1E-4

# the algo's choice
ALGO = Fast

# the kpoint set
KSPACING = 0.2

# 1.3 E_cut of Ni

# not create WAVECAR CHGCAR
LWAVE = .FALSE.

    # 2.3 scnet计算

    任务提交脚本

    #!/bin/bash
#SBATCH -J vasp
#SBATCH -N 1 
#SBATCH --ntasks-per-node=64 
#SBATCH -p wzhcnormal

module purge
source /work/home/if_xie/apprepo/vasp/6.4.1-ioptcell_intelmpi2017_hdf5_libxc/scripts/env.sh

export MKL_DEBUG_CPU_TYPE=5 #加速代码
export MKL_CBWR=AVX2 #使cpu默认支持avx2
export I_MPI_PIN_DOMAIN=numa #内存位置与cpu位置绑定，加速内存读取。对于内存带宽要求高的计算提速明显

# Dirrectory containing the jobs
WORK_DIR=$(pwd)

# Paths to the INCAR and POTCAR files
INCAR_PATH="$WORK_DIR/../INCAR"
POTCAR_PATH="$WORK_DIR/../POTCAR"

# List of job directories
JOB_DIRS=$(ls POSCAR_*)

for JOB_DIR in $JOB_DIRS; do
    # Go into the job directory
    cd $WORK_DIR
    DIR_NAME="dir_$JOB_DIR"
    mkdir $DIR_NAME
    cd $WORK_DIR/$DIR_NAME

    # Create symbolic links to the INCAR and POTCAR files
    ln -s $INCAR_PATH INCAR
    ln -s $POTCAR_PATH POTCAR
    ln -s $WORK_DIR/$JOB_DIR POSCAR

    srun --mpi=pmi2 vasp_std
done
