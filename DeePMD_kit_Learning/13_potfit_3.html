            # potfit修改EAM势函数的方法

            定义自己的解析势函数需要修改两个src目录下的文件

            # 1. 修改functions.itm文件

            这个文件主要用于声明你所定义的势函数，在任意一行加入下面代码即可:

            FUNCTION(mishin_fcc_eam_pair_potential, 5);
            1

            FUNCTION用于势函数的声明，第一个变量是一个字符串变量作为字符串名称，不要使用引号。第二个变量是待拟合参数个数。

            注意：名称不能重复，已有默认的70多个函数名称，其中还有一个就叫mishin，是Mishin用于描述电荷密度的函数。

            # 2. 修改functions_impl.c文件

            新定义一个void函数，名称是设置名称在加上后缀_value

            例子如下所示：

            void mishin_fcc_eam_pair_potential_value(const double r, const double* p, double* f)
{
  double x[2] = {r / p[3], r / p[3]};
  double y[2] = {p[1], p[2]};
  double power[2] = {0, 0};

  power_m(2, power, x, y);

  *f = p[0] / (p[2] - p[1]) * (p[2] / power[0] - p[1] / power[1]) + p[4];
}

            # 3. EAM势函数的定义

            根据Mishin在这里 (opens new window)提出的工作，EAM的三个部分如下所示：

            ϕ(r)=ψ(r−rch)V0b2−b1(b2(r/r1)b1−b1(r/r1)b2)ρ(r)=ψ(r−rch)[A0(r−r0)yexp(−γ(r−r0))(1+B0exp(−γ(r−r0)))+C0]F(ρ)=F0(1−γln(ρ))ργ+F1ρ

            函数的具体内容如下所示：

            void zhou_psi_r_value(const double r, const double* p, double* f)
{
  double z = r / p[0] - 1;
  double temp1 = exp(-p[1] * z);
  double temp2 = exp(-p[2] * z);
  double temp3 = r / p[0] - p[5];
  double temp4 = r / p[0] - p[6];
  double power1 = 0;
  double power2 = 0;
  double index1 = 20.0;
  power_1(&power1, &temp3, &index1);
  power_1(&power2, &temp4, &index1);

  *f = p[3] * temp1 / (1 + power1) - p[4] * temp2 / (1 + power2);
}

void zhou_rho_r_value(const double r, const double* p, double* f)
{
  double z = r / p[1] - 1;
  double temp = exp(-p[2] * z);
  double power1 = 0;
  double tmp = r / p[1] -p[3];
  double index1 = 20.0;
  power_1(&power1, &tmp, &index1);

  *f = p[0] * temp / (1 + power1);
}

void zhou_F_rho_value(const double r, const double* p, double* f)
{
  double rho_n = 0.85 * p[0];
  double rho_o = 1.15 * p[0];
  double x1 = r / rho_n - 1;
  double x2 = r / p[0] - 1;
  double x3 = r / p[1];
  double index1 = 2.0;
  double index2 = 3.0;
  if (r < rho_n){
    double power1 = 0;
    double power2 = 0;
    power_1(&power1, &x1, &index1);
    power_1(&power2, &x1, &index2);
    *f = p[2] + p[3] * x1 + p[4] * power1 + p[5] * power2;
  }
  
  if (r >= rho_n && r < rho_o){
    double power3 = 0;
    double power4 = 0;
    power_1(&power3, &x2, &index1);
    power_1(&power4, &x2, &index2);
    *f = p[6] + p[7] * x2 + p[8] * power3 + p[9] * power4;
  }

  if (r >= rho_o){
    double power5 = 0;
    power_1(&power5, &x3, &p[11]);
    *f = p[10] * (1 - log(power5)) * power5;
  }
}

            # 5. 并行拟合

            由于Intel编译器的并行包我安装过，Potfit也提供了并行版本的编译，具体过程如下所示：

            source ~/intel/oneapi/setvars.sh
./waf configure -i eam -m apot  --enable-mpi
./waf build
#在bin目录下出现potfit_apot_eam_mkl_mpi
            1
            2
            3
            4
            nohup mpirun -np 64 ./potfit_apot_eam_mkl_mpi Ni_eam.param > output.log 2>&1 &
            1

            测试成功，对于680个Ni的fcc超晶胞构型（108原子）运行上述代码用时34分钟。

            但是最终的结果十分奇怪（结果过于丑陋，单调性什么的都不对，就不展示了），我认为可能的原因有：

            1. 初始参数设置的不合理；（没有进行试参工作，并且初始参数设置可能不合理）

            2. 没有完全仿照Mishin的工作；（和状态方程对比这部分难以代码实现，而且没有用到数据集）

            我认为Zhou势函数有效地改善了这个问题，因此换用zhou势函数再次尝试。
