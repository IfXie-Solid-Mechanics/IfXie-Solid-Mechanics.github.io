    # 1.1 整理OUTCAR文件

    从scnet上下载Fe_training_data和Ni_training_data等训练数据至ZHANG01的vasp_training_data目录后，通过下面的copy_data.sh脚本将OUTCAR文件集中

    #!/bin/bash

# 启用globstar以支持**/递归匹配
shopt -s globstar

# 定义目标目录，用来存放所有的OUTCAR文件
output_dir="./all_outcar_files"

mkdir -p $output_dir

# 初始化计数器
count=0

# 遍历Fe_training_data目录下的所有子目录，查找OUTCAR文件
for dir in ./Fe_training_data/**/; do
  if [ -f "$dir/OUTCAR" ]; then
    echo "Found OUTCAR in: $dir"

    # 调试输出，检查路径
    echo "Original directory path: $dir"

    # 去掉路径中的 './' 前缀，然后替换 '/' 为 '_'
    safe_name=$(echo "$dir" | sed 's|^\./||' | tr '/' '_')
    echo "Safe file name: $safe_name"

    # 复制文件
    cp "$dir/OUTCAR" "$output_dir/${safe_name}_OUTCAR"

    # 增加计数
    count=$((count + 1))
  fi
done

    # 1.2 调用dpdata对于OUTCAR进行格式转化

    下面是张寅老师放在GitHub上的代码https://github.com/yzhang951/MD-utility/blob/main/DeePMD/data_conv.py

    import dpdata
import numpy as np

vasp_multi_systems = dpdata.MultiSystems.from_dir(
    dir_name="./", file_name="*OUTCAR", fmt="vasp/outcar"
)
training_set = dpdata.MultiSystems()
validation_set = dpdata.MultiSystems()
#print(vasp_multi_systems.systems)
#vasp_multi_systems.to_deepmd_raw("./deepmd_data/")
#vasp_multi_systems.to_deepmd_npy("./deepmd_data/")

for system in vasp_multi_systems:
    frame_num = len(system)
    index_validation = np.random.choice(frame_num, size=int(0.2*frame_num), replace=False)
    index_training = list(set(range(frame_num))-set(index_validation))

    data_training = system.sub_system(index_training)
    training_set.append(data_training)
    if len(index_validation)>0:
        data_validation = system.sub_system(index_validation)
        validation_set.append(data_validation)
#    print('# the training data contains %d frames' % len(index_training))
#    print('# the validation data contains %d frames' % len(index_validation))

print(training_set)
print(validation_set)
training_set.to_deepmd_raw('training_data')
training_set.to_deepmd_npy('training_data')
validation_set.to_deepmd_raw('validation_data')
validation_set.to_deepmd_npy('validation_data')

    # 1.3 存在未收敛结构时dpdata报错

    说明，如果存在未收敛的结构，调用上述命令会报错，报错主要信息如下：

    /home/ifxie/anaconda3/envs/deepmd/lib/python3.10/site-packages/dpdata/vasp/outcar.py:102:
UserWarning: The following structures were unconverged: [1]; so they are not collected.
    1
    2

    这个时候需要对于OUTCAR文件进行检查，使用如下命令即可：

    grep "unconverged" *OUTCAR
    1

    然后我检查出了确实存在3个构型没有收敛，在超算上增加电子步上限重算即可

    Ni_training_data_result_Ni_300K_distorted_10_dir_POSCAR-026__OUTCAR:------------------------ aborting loop EDIFF was not reached (unconverged)  ----------------------------
Ni_training_data_result_Ni_300K_distorted_5_dir_POSCAR-021__OUTCAR:------------------------ aborting loop EDIFF was not reached (unconverged)  ----------------------------
Ni_training_data_result_Ni_300K_distorted_6_dir_POSCAR-023__OUTCAR:------------------------ aborting loop EDIFF was not reached (unconverged)  ----------------------------
    1
    2
    3

    保险起见，以后将NELM都调为200，确保收敛。
