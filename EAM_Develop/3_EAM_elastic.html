          # 基于EAM势函数计算弹性模量，晶格常数

          # 1. Born-Cauchy Rule

          重要性：BC定理是在0K下，连接原子尺度模型和连续介质模型的桥梁。其分为两个部分，这里先简单地介绍针对纯金属的部分。

          对于纯金属而言，其原子排布在具有周期性的晶格点位上。所以金属的原子模型可以通过将一个原子，通过三个晶胞方向不断周期性地平移得到。

          对于连续介质模型，我们通常通过对于应变能密度w进行积分求出应变能W。而应变能密度是位移梯度的函数。然而对于原子尺度的模型，我们不能取一个微观上无穷小的模型来分析应变能密度，因为很可能你分出来的微元中根本没有原子。

          所以Cauchy-Born Rule给出了对于原子尺度局部应变能的计算公式：对于2体势函数 V2(r)，对于宏观上的位移梯度 A=∇u，应变能密度为 $ w = \frac{1}{2V_{0}}\sum_{\mathbf{s}} V_{2}((\mathbf{I}+\mathbf{A})\cdot\mathbf{s})，其中V_{0}是变形前的元胞体积，\mathbf{s}是变形前的晶格。对于3体势，则有w=\frac{1}{3!V_{0}}\sum_{\mathbf{s_{1}},\mathbf{s_{2}}} V_{3}((\mathbf{I}+\mathbf{A})\cdot\mathbf{s_{1}},(\mathbf{I}+\mathbf{A})\cdot\mathbf{s_{2}})$

          所以从势函数模型，可以计算出应变能。根据应变能的局部最小值即可计算出晶格常数，根据应变能在平衡位置附近的二阶导数即可计算出弹性模量。

          再次强调这是对于纯金属简单晶格的理论，如果是合金，Cauchy-Born Rule中提出了能量最小化约束，这也符合晶格畸变的现象。

          # 2. EAM势函数计算晶格常数

          # 2.1 理论部分

          在1984年第一篇提出EAM势函数的文章中，便直接给出了EAM计算晶格常数的公式(DOI: 10.1103/PhysRevB.29.6443)

          通过Cauchy-Born Rule便可以简单地证明文章中(6),(7a),(7b)的正确性。

          # 2.2 代码实现

          在代码实现方面，我们需要先建模获得一个较大的fcc的POSCAR文件，box的大小需要大于势函数的截断半径。然后将这个POSCAR的原子坐标部分进行复制，再进行一定的特殊处理：将(0,0,0)改为中心原子，并且对于坐标进行归一化处理(具体的晶格常数在代码中指定)。获得一个名为fcc-origin.txt的文本文件，文件在附录1中可以看见。

          我们通过DFT已经获得了Ni的理论晶格常数约为3.52埃，此时便可以编程获得附录2中的代码，将DFT的信息和上述理论相结合，并以代码的方式计算target_func

          # 3. EAM势函数计算弹性常数

          # 3.1 理论部分

          在1984年第一篇提出EAM势函数的文章中，也直接给出了EAM计算弹性常数的公式(DOI: 10.1103/PhysRevB.29.6443)

          这里通过Cauchy-Born Rule的处理是类似的，不过就是再求一阶导数，这里将不再重复这一冗杂的工作。

          # 3.2 代码实现

          几乎和晶格常数计算思路一致，在附录3中给出了11方向的代码实现。唯一不同的是，晶格常数是以应变能一阶导数与0的差值构建的target_function。而弹性常数部分则是根据EAM计算得到的弹性常数，与DFT中计算得到的弹性常数的差值来构建的target_function。（至于最终参考是DFT还是实验应该会在后续的章节中讨论）所以在代码中只给出了C11的计算结果，还没构造target_func

          注：其中的F和rho的高阶求导确实极为复杂，我是通过python中的sympy库来辅助完成的。

omiga=4/3 * math.pi * (math.sqrt(2)/4 * a) ** 3 # single atom's volume in fcc (A^3)

f_e=2.007018                                    # Zhou_Ni

#-----function psi-----
def psi(r, A, B, beta, kappa):
    alpha=1.875 * beta
    lambda_psi=2 * kappa
    result=(A * math.exp(-alpha * (r/r_e - 1)))/(1 + (r/r_e - kappa) ** 20) - (B * math.exp(-beta * (r/r_e - 1)))/(1 + (r/r_e - lambda_psi) ** 20)
    return result

#-----function psi's 1st derivation-----
def psi_1_st_derivation(r, A, B, beta, kappa):
    alpha=1.875 * beta
    lambda_psi=2 * kappa
    result=-A*alpha*math.exp(-alpha*(r/r_e - 1))/(r_e*((-kappa + r/r_e)**20 + 1)) - 20*A*(-kappa + r/r_e)**19*math.exp(-alpha*(r/r_e - 1))/(r_e*((-kappa + r/r_e)**20 + 1)**2) + B*beta*math.exp(-beta*(r/r_e - 1))/(r_e*((-lambda_psi + r/r_e)**20 + 1)) + 20*B*(-lambda_psi + r/r_e)**19*math.exp(-beta*(r/r_e - 1))/(r_e*((-lambda_psi + r/r_e)**20 + 1)**2)
    return result

#-----function psi's 2nd derivation-----
def psi_2_nd_derivation(r, A, B, beta, kappa):
    alpha=1.875 * beta
    lambda_psi=2 * kappa
    result=A*alpha**2*math.exp(-alpha*(r/r_e - 1))/(r_e**2*((-kappa + r/r_e)**20 + 1)) + 40*A*alpha*(-kappa + r/r_e)**19*math.exp(-alpha*(r/r_e - 1))/(r_e**2*((-kappa + r/r_e)**20 + 1)**2) + 800*A*(-kappa + r/r_e)**38*math.exp(-alpha*(r/r_e - 1))/(r_e**2*((-kappa + r/r_e)**20 + 1)**3) - 380*A*(-kappa + r/r_e)**18*math.exp(-alpha*(r/r_e - 1))/(r_e**2*((-kappa + r/r_e)**20 + 1)**2) - B*beta**2*math.exp(-beta*(r/r_e - 1))/(r_e**2*((-lambda_psi + r/r_e)**20 + 1)) - 40*B*beta*(-lambda_psi + r/r_e)**19*math.exp(-beta*(r/r_e - 1))/(r_e**2*((-lambda_psi + r/r_e)**20 + 1)**2) - 800*B*(-lambda_psi + r/r_e)**38*math.exp(-beta*(r/r_e - 1))/(r_e**2*((-lambda_psi + r/r_e)**20 + 1)**3) + 380*B*(-lambda_psi + r/r_e)**18*math.exp(-beta*(r/r_e - 1))/(r_e**2*((-lambda_psi + r/r_e)**20 + 1)**2)
    return result

    return result

    return result


#-----function F-----
def F(rho, F_e, neta):

    F_3=F_e * (neta ** 2)/0.45 * (1.15 ** (neta ** -2)) * (math.log(1.15) * (1.15/0.15 - neta + 1) - 1)
    F_2=10/3 * (-F_e * (neta ** 2) * (1.15 ** (neta ** -1))) * math.log(1.15) - 3 * F_3 * 1.3225
    F_1=0
    F_0=F_e * (1 - neta * math.log(1.15)) * 1.15 ** neta - F_2 * 0.0225 - F_3 * 0.003375
    F_n0=F_0 + F_2 * 0.0225 - F_3 * 0.003375
    F_n1=0.85 * (-0.3 * F_2 + 0.0675 * F_3)
    F_n2=0.7225 * (F_2 - 0.45 * F_3)
    F_n3=F_n0 + F_n2 - F_n1

    if rho<0.85 * rho_e:
        rho_n = 0.85 * rho_e
        result=F_n0 + F_n1 * (rho/rho_n - 1) + F_n2 * (rho/rho_n - 1) ** 2 + F_n3 * (rho/rho_n - 1) ** 3
        return result
    if rho<1.15 * rho_e:
        result=F_0 + F_1 * (rho/rho_e - 1) + F_2 * (rho/rho_e - 1) ** 2 + F_3 * (rho/rho_e - 1) ** 3
        return result
    result=F_e * (1 - neta * math.log(rho/rho_e)) * (rho/rho_e) ** neta
    return result

#-----function F's 1st derivation-----
def F_1st_derivation(rho, F_e, neta):

    F_3=F_e * (neta ** 2)/0.45 * (1.15 ** (neta ** -2)) * (math.log(1.15) * (1.15/0.15 - neta + 1) - 1)
    F_2=10/3 * (-F_e * (neta ** 2) * (1.15 ** (neta ** -1))) * math.log(1.15) - 3 * F_3 * 1.3225
    F_1=0
    F_0=F_e * (1 - neta * math.log(1.15)) * 1.15 ** neta - F_2 * 0.0225 - F_3 * 0.003375
    F_n0=F_0 + F_2 * 0.0225 - F_3 * 0.003375
    F_n1=0.85 * (-0.3 * F_2 + 0.0675 * F_3)
    F_n2=0.7225 * (F_2 - 0.45 * F_3)
    F_n3=F_n0 + F_n2 - F_n1

    if rho<0.85 * rho_e:
        rho_n = 0.85 * rho_e
        result=F_n1/rho_n + 2 * F_n2/rho_n * (rho/rho_n - 1) + 3 * F_n3/rho_n * (rho/rho_n - 1) ** 2
        return result
    if rho<1.15 * rho_e:
        result=F_1/rho_e + 2 * F_2/rho_e * (rho/rho_e - 1) + 3 * F_3/rho_e * (rho/rho_e - 1) ** 2
        return result
    result=(-F_e/rho_e) * neta * neta * math.log(rho/rho_e) * (rho/rho_e) ** (neta - 1)
    return result

#-----function F's 2nd derivation-----
def F_2nd_derivation(rho, F_e, neta):

    F_3=F_e * (neta ** 2)/0.45 * (1.15 ** (neta ** -2)) * (math.log(1.15) * (1.15/0.15 - neta + 1) - 1)
    F_2=10/3 * (-F_e * (neta ** 2) * (1.15 ** (neta ** -1))) * math.log(1.15) - 3 * F_3 * 1.3225

    F_n0=F_0 + F_2 * 0.0225 - F_3 * 0.003375
    F_n1=0.85 * (-0.3 * F_2 + 0.0675 * F_3)
    F_n2=0.7225 * (F_2 - 0.45 * F_3)
    F_n3=F_n0 + F_n2 - F_n1

    if rho<0.85 * rho_e:
        rho_n = 0.85 * rho_e

        return result
    if rho<1.15 * rho_e:
        result=2 * F_2/(rho_e ** 2) + 6 * F_3/(rho_e ** 2) * (rho/rho_e - 1)
        return result
    result=-F_e * (neta ** 2) / (rho_e ** 2) * (rho/rho_e) ** (neta - 2) - F_e * (neta ** 2) * (neta - 1) / (rho_e ** 2) * math.log(rho/rho_e) * (rho/rho_e) ** (neta - 2)
    return result

        result=F_n0 + F_n1 * (rho/rho_n - 1) + F_n2 * (rho/rho_n - 1) ** 2 + F_n3 * (rho/rho_n - 1) ** 3
        return result
    if rho<1.15 * rho_e:
        result=F_0 + F_1 * (rho/rho_e - 1) + F_2 * (rho/rho_e - 1) ** 2 + F_3 * (rho/rho_e - 1) ** 3
        return result
    result=F_e * (1 - neta * math.log(rho/rho_e)) * (rho/rho_e) ** neta
    return result

#-----function F's 1st derivation-----
def F_1st_derivation(rho, F_e, neta):

    F_3=F_e * (neta ** 2)/0.45 * (1.15 ** (neta ** -2)) * (math.log(1.15) * (1.15/0.15 - neta + 1) - 1)
    F_2=10/3 * (-F_e * (neta ** 2) * (1.15 ** (neta ** -1))) * math.log(1.15) - 3 * F_3 * 1.3225
    F_1=0
    F_0=F_e * (1 - neta * math.log(1.15)) * 1.15 ** neta - F_2 * 0.0225 - F_3 * 0.003375
    F_n0=F_0 + F_2 * 0.0225 - F_3 * 0.003375
    F_n1=0.85 * (-0.3 * F_2 + 0.0675 * F_3)
    F_n2=0.7225 * (F_2 - 0.45 * F_3)
    F_n3=F_n0 + F_n2 - F_n1

    if rho<0.85 * rho_e:
        rho_n = 0.85 * rho_e
        result=F_n1/rho_n + 2 * F_n2/rho_n * (rho/rho_n - 1) + 3 * F_n3/rho_n * (rho/rho_n - 1) ** 2
        return result
    if rho<1.15 * rho_e:
        result=F_1/rho_e + 2 * F_2/rho_e * (rho/rho_e - 1) + 3 * F_3/rho_e * (rho/rho_e - 1) ** 2
        return result
    result=(-F_e/rho_e) * neta * neta * math.log(rho/rho_e) * (rho/rho_e) ** (neta - 1)
    return result

#-----function F's 2nd derivation-----
def F_2nd_derivation(rho, F_e, neta):

    F_3=F_e * (neta ** 2)/0.45 * (1.15 ** (neta ** -2)) * (math.log(1.15) * (1.15/0.15 - neta + 1) - 1)
    F_2=10/3 * (-F_e * (neta ** 2) * (1.15 ** (neta ** -1))) * math.log(1.15) - 3 * F_3 * 1.3225
    F_1=0
    F_0=F_e * (1 - neta * math.log(1.15)) * 1.15 ** neta - F_2 * 0.0225 - F_3 * 0.003375
    F_n0=F_0 + F_2 * 0.0225 - F_3 * 0.003375
    F_n1=0.85 * (-0.3 * F_2 + 0.0675 * F_3)
    F_n2=0.7225 * (F_2 - 0.45 * F_3)
    F_n3=F_n0 + F_n2 - F_n1

    if rho<0.85 * rho_e:
        rho_n = 0.85 * rho_e
        result=2 * F_n2/(rho_n ** 2) + 6 * F_n3/(rho_n ** 2) * (rho/rho_n - 1)
        return result
    if rho<1.15 * rho_e:
        result=2 * F_2/(rho_e ** 2) + 6 * F_3/(rho_e ** 2) * (rho/rho_e - 1)
        return result
    result=-F_e * (neta ** 2) / (rho_e ** 2) * (rho/rho_e) ** (neta - 2) - F_e * (neta ** 2) * (neta - 1) / (rho_e ** 2) * math.log(rho/rho_e) * (rho/rho_e) ** (neta - 2)
    return result
