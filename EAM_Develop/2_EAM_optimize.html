        # EAM势函数的优化方法

        # 1. 单纯形法

        单纯形法，也称为Nelder-Mead Algorithm。

        这个方法的特点在于“无梯度优化”，虽然其算法复杂度较高，但是用于待优化参数较少的场景下仍然具有不错的表现。

        教材中的其他描述： The Nelder-Mead Algorithm is robust even when the function is discontinuous.

        # 1.1 原理介绍

        一个简单地理解：例如在二维空间中，三角形就是一个单纯形，将这个三角形面积不断缩小，最终收敛至二维空间上的一个点，表示这个点的两个坐标就是最终优化得到的参数

        # 1.1.1 Nelder-Mead Algorithm的初始化

        对于n个待优化参数，先通过算法生成一个n+1维的单纯形。（例如在二维平面上生成一个三角形，在三维空间中生成一个四面体）

        更具体地说，就是对于n个待优化参数，在估计值周围选取n+1个n维向量，并且保证这n+1个向量不出现任意三者共线的情况。

        假设我们待优化的方程为f(x)，而创建的单纯形的n+1个顶点代入优化方程中排序，从小到大标号:f(x1)<=f(x2)<=f(x3)<=...<=f(xn+1)，其中xn+1被称为the worst point.

        注：附录1中的代码，更希望你给出一个初始的比较合理的单纯形，虽然我也设置了initial_simplex=None是可以运行程序的，但是默认版本参数范围较为固定（比如幂次和乘数的范围应该不同）。在附录2中，给出了一个根据各个参数的估计值，得到一个合理的单纯形初始值的代码。

        # 1.1.2 Nelder-Mead Algorithm的优化

        计算除了the worst point以外的重心 x0=1n∑i=1nxi

        根据重心和最差点计算出反射点（reflection point）xr=x0+α(x0−xn+1) ，通常α=1。

        计算发射点处的函数值，可能会出现下面三种情况：

        1. f(x1)<=f(xr)<=f(xn) ，即反射点不是最优的也不是最差的；

        2. f(xr)<=f(x1) ,即反射点是最优的；

        3. f(xn)<=f(xr) ,即反射点是最差的。

        假设出现了情况1，那么将会用反射点代替原先的最差点，生成一个新的单纯性继续迭代优化。

        假设出现了情况2，说明往这个方向优化参数是一个不错的行为，那么需要计算拓展点(expend point): xe=x0+β(x0−xn+1) ，通常β=2。计算出来的这个拓展点可能出现情况2.1和情况2.2

        情况2.1 f(xe)<=f(xr) ，取拓展点代替原先的最差点，生成一个新的单纯性继续迭代优化。

        情况2.2 f(xr)<=f(xe) ，取反射点代替原先的最差点，生成一个新的单纯性继续迭代优化。

        假设出现了情况3，那么说明这一步优化走得很差，需要计算内聚点(contraction point): xc=x0+γ(xn+1−x0) ，通常γ=0.5。计算出来的这个内聚点可能出现情况3.1和情况3.2

        情况3.1 f(xc)<=f(xn+1)，取内聚点代替原先的最差点，生成一个新的单纯性继续迭代优化。

        情况3.2 f(xn+1)<=f(xc)，进行新的单纯形压缩构造(shrink)： xi=x1+δ(xi−x1),通常δ=0.5，即把单纯形靠着最小点附近收缩，形成新的单纯形继续优化。

        # 1.1.3 Nelder-Mead Algorithm的迭代终止条件

        优化结果单纯形中的最大值和最小值之差小于tolerance即可（单纯形面积收缩到了某一点）。或者达到了max_iterations，防止死循环。

        # 1.2 代码调用

        objective_function是优化方程，initial_simplex是初始的单纯形（推荐通过附录2的代码生成一个），以及注明单纯形的维度n_dim，默认是2维。计算反射点、拓展点、内聚点、收缩单纯形的固定参数我都按照默认值设置好了，但是也可以根据自己的需求修改。误差限tolerance和最大优化步骤max_iterations设置了默认值，这个也可以根据自己需求修改。

        # 附录1-单纯形法具体代码

        以Rosenbrock func的优化为例子

        import numpy as np

class NelderMead:
    def __init__(self, objective_function, initial_simplex=None, n_dim=2,
                 alpha=1.0, beta=2.0, gamma=0.5, delta=0.5,
                 max_iterations=10000, tolerance=1e-6):
        """
        init Nelder-Mead
        
        Parameters:
            objective_function: target function,accept a numpy and return a double
            initial_simplex: initial simplex list with (n+1) points, its size is (n+1, n)
            n_dim: the dimension of the problem, default is 2
            alpha: reflection para, default is 1.0
            beta: expension para, default is 2.0
            gamma: shrink para, default is 0.5
            delta: contraction para, default is 0.5
            max_iterations: the max iteration steps, default is 10000
            tolerance: stop condition, default is 1e-6
        """
        self.objective_function = objective_function

        if initial_simplex is None:
            # if the initial simplex is none, create one
            self.n_dim = n_dim
            bounds = [(-5.0, 5.0)] * n_dim  
            self.simplex = np.array([self._generate_initial_point(bounds) for _ in range(n_dim + 1)])
        else:
            # initial the simplex
            self.simplex = initial_simplex
            self.n_dim = self.simplex.shape[1]

        # set para
        self.alpha = alpha
        self.beta = beta
        self.gamma = gamma
        self.delta = delta
        self.max_iterations = max_iterations
        self.tolerance = tolerance
        
        # the best point and its position
        self.best_value = None
        self.best_point = None

    def _generate_initial_point(self, bounds):
        """
        generate_initial_point(if the initial simplex is none)
        """
        return np.array([np.random.uniform(low=bounds[i][0], high=bounds[i][1]) for i in range(len(bounds))])
    
    def compute_objective_values(self):
        """
        cal the simplex value for each points, and sort it from small value to large value
        """
        #cal
        values = np.array([self.objective_function(point) for point in self.simplex])
        
        # sort
        sorted_indices = np.argsort(values)
        self.simplex = self.simplex[sorted_indices]
        values = values[sorted_indices]
        
        return values
    
    def iterate(self):
        """
        one iterate cal
        """
        n_dim = self.n_dim
        
        # cal objective function and sort the simplex
        values = self.compute_objective_values()
        print(values)
        # get the best point 
        best_point = self.simplex[0]
        if self.best_value is None or values[0] < self.best_value:
            self.best_value = values[0]
            self.best_point = best_point.copy()
        
        # cal the reflection middle x_0
        x_0=self.simplex[0]
        for i in range(1,n_dim):
            x_0=x_0 + self.simplex[i]
        x_0=x_0/n_dim

        # cal the reflection point x_r
        x_r = self._compute_reflection(x_0)
        value_r = self.objective_function(x_r)
        
        if value_r < values[0]:
            # cal the expansion
            x_e = self._compute_expansion(x_r, x_0)
            value_e = self.objective_function(x_e)
            
            if value_e < value_r:
                # x_e is better
                self.simplex[n_dim] = x_e.copy()
            else:
                # x_e is worse
                self.simplex[n_dim] = x_r.copy()

        elif value_r < values[(n_dim - 1)]:
            # x_r is not enough good, but not too bad
            self.simplex[n_dim] = x_r.copy()
        else:
            # x_r is still the worst, we need to contract
            x_c = self._compute_contraction(x_0)
            value_c = self.objective_function(x_c)
            if value_c < values[n_dim]:
                self.simplex[n_dim] = x_c.copy()
            else: 
                # x_c is still the worst, we need to shrink the simplex
                for i in range(1,n_dim+1):
                    self.simplex[i] = x_0 + self.delta * (self.simplex[i] - x_0)
            
            
    def _compute_reflection(self, x_0):
        """
        compute reflection point
        """
        reflection_point = x_0 + self.alpha * (x_0 - self.simplex[self.n_dim])
        return reflection_point
    
    def _compute_expansion(self, x_r, x_0):
        """
        compute expansion point
        """
        expansion_point = x_0 + self.beta * (x_r - x_0)
        return expansion_point
    
    def _compute_contraction(self, x_0):
        """
        compute contraction point
        """
        contraction_point = x_0 + self.gamma * (self.simplex[self.n_dim] - x_0)
        return contraction_point
    
    def optimize(self):
        """
        optimize control function
        """
        for _ in range(self.max_iterations):
            # cal 
            values = self.compute_objective_values()
            
            # determine iterate or not
            if np.ptp(values) < self.tolerance:
                print("eventual values:", values)
                break
            
            # iterate
            self.iterate()
        print("achieve max iterations")
        return self.best_point, self.best_value

# test function: Rosenbrock func
def rosenbrock(x):
    a = 1.0
    b = 100.0
    return (a - x[0])**2 + b*(x[1] - x[0]**2)**2

# initial optimizer
n_dim = 2
optimizer = NelderMead(objective_function=rosenbrock, n_dim=n_dim,
                      max_iterations=1000, tolerance=1e-6)

# optimize
best_point, best_value = optimizer.optimize()

print("Optimization complete!")
print(f"Best point: {best_point}")
print(f"Best value: {best_value:.6f}")

        # 附录2-初始单纯形的生成

        这个python代码要求首先需要指定参数的个数n，并且指定每个参数的范围para_range[i]，这些内容从文件para.txt中读取，第一行是n，后面的n行每一行分别对应para_range[i]的下限与上限。然后生成一个n+1维的单纯形initial_simplex。

        import numpy as np

def generate_initial_simplex(n, para_range):
    """
    init simplex
    
    parameters:
    n - dimension
    para_range - range list of every para of n dimensions，every element is (low, high).
    
    return:
    initial_simplex - (n+1, n)'s NumPy 
    """

    # initial
    simplex = []
    
    # create n+1 points
    for _ in range(n + 1):

        point = []
        for i in range(n):
            low, high = para_range[i]
            point.append(np.random.uniform(low, high))
        simplex.append(point)
    
    return np.array(simplex)
    

def is_non_degenerate(simplex):
    """
    check the simplex is not degenerate
    """
    n = simplex.shape[1]  # dimension
    vectors = simplex[1:] - simplex[0]  
    rank = np.linalg.matrix_rank(vectors)
    return rank == n
